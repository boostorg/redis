15 // Arbitrary struct to de/serialize.
16 struct mystruct {
17  std::int32_t x;
18  std::string y;
19 };
20 
21 // Serializes mystruct
22 void to_bulk(std::string& to, mystruct const& obj)
23 {
24  using aedis::resp3::type;
25  using aedis::resp3::add_header;
26  using aedis::resp3::add_separator;
27 
28  auto const size = sizeof obj.x + obj.y.size();
29  add_header(to, type::blob_string, size);
30  auto const* p = reinterpret_cast<char const*>(&obj.x);
31  std::copy(p, p + sizeof obj.x, std::back_inserter(to));
32  std::copy(std::cbegin(obj.y), std::cend(obj.y), std::back_inserter(to));
33  add_separator(to);
34 }

36 // Deserialize the struct.
37 void from_string(mystruct& obj, boost::string_view sv, boost::system::error_code& ec)
38 {
39  char* p = reinterpret_cast<char*>(&obj.x);
40  std::copy(std::cbegin(sv), std::cbegin(sv) + sizeof obj.x, p);
41  std::copy(std::cbegin(sv) + sizeof obj.x, std::cend(sv), std::back_inserter(obj.y));
42 }
