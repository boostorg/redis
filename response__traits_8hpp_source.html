164 template <class Tuple>
165 class flat_transaction_adapter {
166 private:
167  using variant_type =
168  boost::mp11::mp_rename<boost::mp11::mp_transform<response_traits_t, Tuple>, std::variant>;
169 
170  std::size_t i_ = 0;
171  std::size_t aggregate_size_ = 0;
172  std::array<variant_type, std::tuple_size<Tuple>::value> adapters_;
173 
174 public:
175  flat_transaction_adapter(Tuple* r)
176  { assigner<std::tuple_size<Tuple>::value - 1>::assign(adapters_, *r); }
177 
178  void count(type t, std::size_t aggregate_size, std::size_t depth)
179  {
180  if (depth == 1) {
181  if (is_aggregate(t))
182  aggregate_size_ = aggregate_size;
183  else
184  ++i_;
185 
186  return;
187  }
188 
189  if (--aggregate_size_ == 0)
190  ++i_;
191  }
192 
193  void
194  operator()(
195  type t,
196  std::size_t aggregate_size,
197  std::size_t depth,
198  char const* data,
199  std::size_t size,
200  std::error_code& ec)
201  {
202  if (depth == 0) {
203  if (aggregate_size != std::tuple_size<Tuple>::value) {
204  ec = adapter::error::incompatible_tuple_size;
205  return;
206  }
207 
208  return;
209  }
210 
211  std::visit([&](auto& arg){arg(t, aggregate_size, depth, data, size, ec);}, adapters_[i_]);
212  count(t, aggregate_size, depth);
213  }
214 };
