188  void count(resp3::node<boost::string_view> const& nd)
189  {
190  if (nd.depth == 1) {
191  if (is_aggregate(nd.data_type))
192  aggregate_size_ = element_multiplicity(nd.data_type) * nd.aggregate_size;
193  else
194  ++i_;
195 
196  return;
197  }
198 
199  if (--aggregate_size_ == 0)
200  ++i_;
201  }
202 
203  void
204  operator()(
205  resp3::node<boost::string_view> const& nd,
206  boost::system::error_code& ec)
207  {
208  using boost::variant2::visit;
209 
210  if (nd.depth == 0) {
211  auto const real_aggr_size = nd.aggregate_size * element_multiplicity(nd.data_type);
212  if (real_aggr_size != std::tuple_size<Tuple>::value)
213  ec = error::incompatible_size;
214 
215  return;
216  }
217 
218  visit([&](auto& arg){arg(nd, ec);}, adapters_[i_]);
219  count(nd);
220  }
221 };
