25/* Traits class for response objects.
26 *
27 * Provides traits for all supported response types i.e. all STL
28 * containers and C++ buil-in types.
29 */
30template <class ResponseType>
31struct response_traits {
32 using adapter_type = adapter::detail::wrapper<typename std::decay<ResponseType>::type>;
33 static auto adapt(ResponseType& r) noexcept { return adapter_type{&r}; }
34};

64template <class T>
65using adapter_t = typename response_traits<std::decay_t<T>>::adapter_type;

67// Duplicated here to avoid circular include dependency.
68template<class T>
69auto internal_adapt(T& t) noexcept
70 { return response_traits<std::decay_t<T>>::adapt(t); }

72template <std::size_t N>
73struct assigner {
74 template <class T1, class T2>
75 static void assign(T1& dest, T2& from)
76 {
77 dest[N] = internal_adapt(std::get<N>(from));
78 assigner<N - 1>::assign(dest, from);
79 }
80};
81
82template <>
83struct assigner<0> {
84 template <class T1, class T2>
85 static void assign(T1& dest, T2& from)
86 {
87 dest[0] = internal_adapt(std::get<0>(from));
88 }
89};

91template <class Tuple>
92class static_aggregate_adapter {
93private:
94 using adapters_array_type =
95 std::array<
96 boost::mp11::mp_rename<
97 boost::mp11::mp_transform<
98 adapter_t, Tuple>,
99 boost::variant2::variant>,
100 std::tuple_size<Tuple>::value>;
101
102 std::size_t i_ = 0;
103 std::size_t aggregate_size_ = 0;
104 adapters_array_type adapters_;
105
106public:
107 explicit static_aggregate_adapter(Tuple* r = nullptr)
108 {
109 detail::assigner<std::tuple_size<Tuple>::value - 1>::assign(adapters_, *r);
110 }
111
112 void count(resp3::node<boost::string_view> const& nd)
113 {
114 if (nd.depth == 1) {
115 if (is_aggregate(nd.data_type))
116 aggregate_size_ = element_multiplicity(nd.data_type) * nd.aggregate_size;
117 else
118 ++i_;
119
120 return;
121 }
122
123 if (--aggregate_size_ == 0)
124 ++i_;
125 }
126
127 void
128 operator()(
129 resp3::node<boost::string_view> const& nd,
130 boost::system::error_code& ec)
131 {
132 using boost::variant2::visit;
133
134 if (nd.depth == 0) {
135 auto const real_aggr_size = nd.aggregate_size * element_multiplicity(nd.data_type);
136 if (real_aggr_size != std::tuple_size<Tuple>::value)
137 ec = error::incompatible_size;
138
139 return;
140 }
141
142 visit([&](auto& arg){arg(nd, ec);}, adapters_[i_]);
143 count(nd);
144 }
145};

147template <class... Ts>
148struct response_traits<std::tuple<Ts...>>
149{
150 using response_type = std::tuple<Ts...>;
151 using adapter_type = static_aggregate_adapter<response_type>;
152 static auto adapt(response_type& r) noexcept { return adapter_type{&r}; }
153};
