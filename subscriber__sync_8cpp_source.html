26 // Subscribe again everytime there is a disconnection.
27 void event_receiver(connection& conn)
28 {
29  request req;
30  req.push("SUBSCRIBE", "channel");
31 
32  for (;;) {
33  auto ev = conn.receive_event();

35  conn.exec(req);
36  }
37 }
