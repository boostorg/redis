23 net::awaitable<void> echo_loop(tcp_socket socket, std::shared_ptr<connection> db)
24 {
25  try {
26  request req;
27  std::tuple<std::string> resp;
28  std::string buffer;
29 
30  for (;;) {
31  auto n = co_await net::async_read_until(socket, net::dynamic_buffer(buffer, 1024), "\n");
32  req.push("PING", buffer);
33  co_await db->async_exec(req, adapt(resp));
34  co_await net::async_write(socket, net::buffer(std::get<0>(resp)));
35  std::get<0>(resp).clear();
36  req.clear();
37  buffer.erase(0, n);
38  }
39  } catch (std::exception const& e) {
40  std::cout << e.what() << std::endl;
41  }
42 }

44 net::awaitable<void> listener()
