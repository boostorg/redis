27 class receiver : public std::enable_shared_from_this<receiver> {
28 private:
29  std::vector<node> resps_;
30  std::queue<std::weak_ptr<user_session_base>> sessions_;
31 
32 public:
33  void on_message(std::error_code ec, command cmd)
34  {
35  if (ec) {
36  std::cerr << "Error: " << ec.message() << std::endl;
37  return;
38  }
39 
40  switch (cmd) {
41  case command::ping:
42  {
43  if (auto session = sessions_.front().lock()) {
44  session->deliver(resps_.front().data);
45  } else {
46  std::cout << "Session expired." << std::endl;
47  }
48 
49  sessions_.pop();
50  } break;
51  case command::incr:
52  {
53  std::cout << "Echos so far: " << resps_.front().data << std::endl;
54  } break;
55  default: { /* Ignore */; }
56  }
57 
58  resps_.clear();
59  }
60 
61  auto get_extended_adapter()
62  {
63  return [adapter = adapt(resps_)](command, type t, std::size_t aggregate_size, std::size_t depth, char const* data, std::size_t size, std::error_code& ec) mutable
64  { return adapter(t, aggregate_size, depth, data, size, ec); };
65  }
66 
67  void add_user_session(std::shared_ptr<user_session_base> session)
68  { sessions_.push(session); }
69 };

81 net::awaitable<void> listener()
82 {
83  auto ex = co_await net::this_coro::executor;
84  net::ip::tcp::acceptor acceptor(ex, {net::ip::tcp::v4(), 55555});
85 
86  auto recv = std::make_shared<receiver>();
87  auto on_db_msg = [recv](std::error_code ec, command cmd)
88  { recv->on_message(ec, cmd); };
89 
90  auto db = std::make_shared<client>(ex);
91  db->set_extended_adapter(recv->get_extended_adapter());
92  db->set_msg_callback(on_db_msg);
93  net::co_spawn(ex, connection_manager(db), net::detached);
94 
95  for (;;) {
96  auto socket = co_await acceptor.async_accept(net::use_awaitable);
97  auto session = std::make_shared<user_session>(std::move(socket));
98 
99  auto on_user_msg = [db, recv, session](std::string const& msg)
100  {
101  db->send(command::ping, msg);
102  db->send(command::incr, "echo-counter");
103  recv->add_user_session(session);
104  };
105 
106  session->start(on_user_msg);
107  }
108 }
