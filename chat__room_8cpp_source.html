26 // Chat over redis pubsub. To test, run this program from different
27 // terminals and type messages to stdin. You may also want to run
28 //
29 // $ redis-cli
30 // > monitor
31 //
32 // To see the message traffic.
33 
34 net::awaitable<void> subscriber(std::shared_ptr<connection> db)
35 {
36  request req;
37  req.push("SUBSCRIBE", "chat-channel");
38 
39  for (response_type resp;;) {
40  auto const ev = co_await db->async_receive_event(aedis::adapt(resp));
41  switch (ev) {
42  case connection::event::push:
43  print_push(resp);
44  break;
45 
46  case connection::event::hello:
47  co_await db->async_exec(req);
48  break;
49 
50  default:;
51  }
52  resp.clear();
53  }
54 }
55 
56 net::awaitable<void>
57 publisher(net::posix::stream_descriptor& in, std::shared_ptr<connection> db)
58 {
59  for (std::string msg;;) {
60  std::size_t n = co_await net::async_read_until(in, net::dynamic_buffer(msg, 1024), "\n", net::use_awaitable);
61  request req;
62  req.push("PUBLISH", "chat-channel", msg);
63  co_await db->async_exec(req);
64  msg.erase(0, n);
65  }
66 }

68 int main()
69 {
70  try {
71  net::io_context ioc{1};
72  net::posix::stream_descriptor in{ioc, ::dup(STDIN_FILENO)};
73 
74  auto db = std::make_shared<connection>(ioc);
75  db->get_config().enable_events = true;
76 
77  co_spawn(ioc, publisher(in, db), net::detached);
78  co_spawn(ioc, subscriber(db), net::detached);
