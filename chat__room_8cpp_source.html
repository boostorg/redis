32// Chat over redis pubsub. To test, run this program from different
33// terminals and type messages to stdin. Use
34//
35// $ redis-cli monitor
36//
37// to monitor the message traffic.
38
39// Receives messages from other users.
40net::awaitable<void> push_receiver(std::shared_ptr<connection> conn)
41{
42 for (std::vector<node<std::string>> resp;;) {
43 co_await conn->async_receive(adapt(resp));
44 print_push(resp);
45 resp.clear();
46 }
47}

49// Subscribes to the channels when a new connection is stablished.
50net::awaitable<void> reconnect(std::shared_ptr<connection> conn)
51{
52 request req;
53 req.get_config().cancel_on_connection_lost = true;
54 req.push("SUBSCRIBE", "chat-channel");
55
56 stimer timer{co_await net::this_coro::executor};
57 endpoint ep{"127.0.0.1", "6379"};
58 for (;;) {
59 boost::system::error_code ec1, ec2;
60 co_await (
61 conn->async_run(ep, {}, net::redirect_error(net::use_awaitable, ec1)) &&
62 conn->async_exec(req, adapt(), net::redirect_error(net::use_awaitable, ec2))
63 );
64 std::clog << "async_run: " << ec1.message() << "\n"
65 << "async_exec: " << ec2.message() << std::endl;
66 conn->reset_stream();
67 timer.expires_after(std::chrono::seconds{1});
68 co_await timer.async_wait();
69 }
70}
71
72// Publishes messages to other users.
73net::awaitable<void> publisher(stream_descriptor& in, std::shared_ptr<connection> conn)
74{
75 for (std::string msg;;) {
76 auto n = co_await net::async_read_until(in, net::dynamic_buffer(msg, 1024), "\n");
77 request req;
78 req.push("PUBLISH", "chat-channel", msg);
79 co_await conn->async_exec(req);
80 msg.erase(0, n);
81 }
82}

84auto main() -> int
85{
86 try {
87 net::io_context ioc{1};
88 stream_descriptor in{ioc, ::dup(STDIN_FILENO)};
89
90 auto conn = std::make_shared<connection>(ioc);
91 co_spawn(ioc, publisher(in, conn), net::detached);
92 co_spawn(ioc, push_receiver(conn), net::detached);
93 co_spawn(ioc, reconnect(conn), net::detached);
94
95 net::signal_set signals(ioc, SIGINT, SIGTERM);
96 signals.async_wait([&](auto, auto){ ioc.stop(); });
97
98 ioc.run();
99 } catch (std::exception const& e) {
100 std::cerr << e.what() << std::endl;
101 }
102}
