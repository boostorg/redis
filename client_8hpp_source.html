136 template <class... Ts>
137 void client::send(redis::command cmd, Ts const&... args)
138 {
139  auto const can_write = prepare_next();
140 
141  auto sr = redis::make_serializer(requests_);
142  auto const before = std::size(requests_);
143  sr.push(cmd, args...);
144  auto const after = std::size(requests_);
145  req_info_.front().size += after - before;;
146 
147  if (!has_push_response(cmd)) {
148  commands_.emplace(cmd);
149  ++req_info_.front().cmds;
150  }
151 
152  if (can_write)
153  timer_.cancel_one();
154 }
