116 template <class... Ts>
117 void client::send(redis::command cmd, Ts const&... args)
118 {
119  auto const can_write = prepare_next();
120 
121  auto sr = redis::make_serializer(requests_);
122  auto const before = std::size(requests_);
123  sr.push(cmd, args...);
124  auto const after = std::size(requests_);
125  req_info_.front().size += after - before;;
126 
127  if (!has_push_response(cmd)) {
128  commands_.emplace(cmd);
129  ++req_info_.front().cmds;
130  }
131 
132  if (can_write)
133  timer_.cancel_one();
134 }
