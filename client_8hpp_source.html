113  template <class... Ts>
114  void send(Command cmd, Ts const&... args)
115  {
116  auto const can_write = prepare_next();
117 
118  auto const before = requests_.size();
119  sr_.push(cmd, args...);
120  auto const d = requests_.size() - before;
121  BOOST_ASSERT(d != 0);
122  info_.back().size += d;;
123 
124  if (!has_push_response(cmd)) {
125  commands_.push_back(std::make_pair(cmd, d));
126  ++info_.back().cmds;
127  }
128 
129  if (can_write)
130  wait_write_timer_.cancel_one();
131  }

366  void on_resolve()
367  {
368  // If we are coming from a connection that was lost we have to
369  // reset the socket to a fresh state.
370  socket_ =
371  std::make_shared<AsyncReadWriteStream>(read_timer_.get_executor());

374  void on_connect()
375  {
376  // When we are reconnecting we can't simply call send(hello)
377  // as that will add the command to the end of the queue, we need
378  // it as the first element.
379  if (info_.empty()) {
380  // Either we are connecting for the first time or there are
381  // no commands that were left unresponded from the last
382  // connection. We can send hello as usual.
383  BOOST_ASSERT(requests_.empty());
384  BOOST_ASSERT(commands_.empty());
385  send(Command::hello, 3);
386  return;
387  }
388 
389  if (info_.front().sent) {
390  // There is one request that was left unresponded when we
391  // e.g. lost the connection, since we erase requests right
392  // after writing them to the socket (to avoid resubmission) it
393  // is lost and we have to remove it.
394 
395  // Noop if info_.front().size is already zero, which happens
396  // when the request was successfully writen to the socket.
397  // In the future we may want to avoid erasing but resend (at
398  // the risc of resubmission).
399  requests_.erase(0, info_.front().size);
400 
401  // Erases the commands that were lost as well.
402  commands_.erase(
403  std::begin(commands_),
404  std::begin(commands_) + info_.front().cmds);
405 
406  info_.front().cmds = 0;
407 
408  // Do not erase the info_ front as we will use it below.
409  // info_.erase(std::begin(info_));
410  }
411 
412  // Code below will add a hello to the front of the request and
413  // update info_ and commands_ accordingly.
414 
415  auto const old_size = requests_.size();
416  sr_.push(Command::hello, 3);
417  auto const hello_size = requests_.size() - old_size;;
418 
419  // Now we have to rotate the hello to the front of the request
420  // (Remember it must always be the first command).
421  std::rotate(
422  std::begin(requests_),
423  std::begin(requests_) + old_size,
424  std::end(requests_));
425 
426  // Updates info_.
427  info_.front().size += hello_size;
428  info_.front().cmds += 1;
429 
430  // Updates commands_
431  commands_.push_back(std::make_pair(Command::hello, hello_size));
432  std::rotate(
433  std::begin(commands_),
434  std::prev(std::end(commands_)),
435  std::end(commands_));
436  }
437 
438  // Prepares the back of the queue to receive further commands. If
439  // true is returned the request in the front of the queue can be
440  // sent to the server.
441  bool prepare_next()
442  {
443  if (info_.empty()) {
444  info_.push_back({});
445  return true;
446  }
447 
448  if (info_.front().sent) {
449  // There is a pending response, we can't modify the front of
450  // the vector.
451  BOOST_ASSERT(info_.front().cmds != 0);
452  if (info_.size() == 1)
453  info_.push_back({});
454 
455  return false;
456  }
457 
458  // When cmds = 0 there are only commands with push response on
459  // the request and we are not waiting for any response.
460  return info_.front().cmds == 0;
461  }

463  // Returns true when the next request can be written.
464  bool on_cmd(command_info_type)
465  {
466  BOOST_ASSERT(!info_.empty());
467  BOOST_ASSERT(!commands_.empty());
468 
469  commands_.erase(std::begin(commands_));
470 
471  if (--info_.front().cmds != 0)
472  return false;
473 
474  info_.erase(std::begin(info_));
475 
476  return !info_.empty();
477  }
