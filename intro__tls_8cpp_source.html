31auto main() -> int
32{
33 try {
34 net::io_context ioc;
35
36 net::ssl::context ctx{net::ssl::context::sslv23};
37
38 connection conn{ioc, ctx};
39 conn.next_layer().set_verify_mode(net::ssl::verify_peer);
40 conn.next_layer().set_verify_callback(verify_certificate);
41
42 request req;
43 req.get_config().cancel_on_connection_lost = true;
44 req.push("PING");
45 req.push("QUIT");
46
47 std::tuple<std::string, aedis::ignore> resp;
48 conn.async_exec(req, adapt(resp), logger);
49 conn.async_run({"127.0.0.1", "6379"}, {}, logger);
50
51 ioc.run();
52
53 std::cout << "Response: " << std::get<0>(resp) << std::endl;
54 } catch (std::exception const& e) {
55 std::cerr << "Error: " << e.what() << std::endl;
