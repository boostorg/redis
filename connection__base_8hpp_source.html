62 auto cancel(operation op) -> std::size_t
63 {
64 switch (op) {
65 case operation::exec:
66 {
67 for (auto& e: reqs_) {
68 e->stop = true;
69 e->timer.cancel_one();
70 }
71
72 auto const ret = reqs_.size();
73 reqs_ = {};
74 return ret;
75 }
76 case operation::run:
77 {
78 resv_.cancel();
79 derived().close();
80
81 read_timer_.cancel();
82 check_idle_timer_.cancel();
83 writer_timer_.cancel();
84 ping_timer_.cancel();
85
86 auto point = std::stable_partition(std::begin(reqs_), std::end(reqs_), [](auto const& ptr) {
87 return !ptr->req->get_config().close_on_connection_lost;
88 });
89
90 // Cancel own pings if there are any waiting.
91 std::for_each(point, std::end(reqs_), [](auto const& ptr) {
92 ptr->stop = true;
93 ptr->timer.cancel();
94 });
95
96 reqs_.erase(point, std::end(reqs_));
97 return 1U;
98 }
99 case operation::receive_push:
100 {
101 push_channel_.cancel();
102 return 1U;
103 }
104 default: BOOST_ASSERT(false); return 0;
105 }
106 }

198 void cancel_push_requests()
199 {
200 auto point = std::stable_partition(std::begin(reqs_), std::end(reqs_), [](auto const& ptr) {
201 return !(ptr->written && ptr->req->size() == 0);
202 });
203
204 std::for_each(point, std::end(reqs_), [](auto const& ptr) {
205 ptr->timer.cancel();
206 });
207
208 reqs_.erase(point, std::end(reqs_));
209 }

296 void stage_request(req_info& ri)
297 {
298 write_buffer_ += ri.req->payload();
299 cmds_ += ri.req->size();
300 ri.written = true;

303 void coalesce_requests()
304 {
305 BOOST_ASSERT(write_buffer_.empty());
306 BOOST_ASSERT(!reqs_.empty());
307
308 stage_request(*reqs_.at(0));
309
310 for (std::size_t i = 1; i < std::size(reqs_); ++i) {
311 if (!reqs_.at(i - 1)->req->get_config().coalesce ||
312 !reqs_.at(i - 0)->req->get_config().coalesce) {
313 break;
314 }
315 stage_request(*reqs_.at(i));
316 }
317 }

329 auto expect_role(std::string const& expected) -> bool
330 {
331 if (std::empty(expected))
332 return true;
333
334 resp3::node<std::string> role_node;
335 role_node.data_type = resp3::type::blob_string;
336 role_node.aggregate_size = 1;
337 role_node.depth = 1;
338 role_node.value = "role";
339
340 auto iter = std::find(std::cbegin(response_), std::cend(response_), role_node);
341 if (iter == std::end(response_))
342 return false;
343
344 ++iter;
345 BOOST_ASSERT(iter != std::cend(response_));
346 return iter->value == expected;
347 }
