111  template <class Command, class Key, class ForwardIterator>
112  void push_range2(Command cmd, Key const& key, ForwardIterator begin, ForwardIterator end)
113  {
114  using value_type = typename std::iterator_traits<ForwardIterator>::value_type;
115 
116  if (begin == end)
117  return;
118 
119  auto constexpr size = resp3::bulk_counter<value_type>::size;
120  auto const distance = std::distance(begin, end);
121  resp3::add_header(*request_, 2 + size * distance);
122  resp3::add_bulk(*request_, to_string(cmd));
123  resp3::add_bulk(*request_, key);
124 
125  for (; begin != end; ++begin)
126  resp3::add_bulk(*request_, *begin);
127  }
