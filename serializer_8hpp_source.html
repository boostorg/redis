111  template <class Command, class Key, class ForwardIterator>
112  void push_range2(Command cmd, Key const& key, ForwardIterator begin, ForwardIterator end)
113  {
114  using value_type = typename std::iterator_traits<ForwardIterator>::value_type;
115  using resp3::type;
116 
117  if (begin == end)
118  return;
119 
120  auto constexpr size = resp3::bulk_counter<value_type>::size;
121  auto const distance = std::distance(begin, end);
122  resp3::add_header(*request_, type::array, 2 + size * distance);
123  resp3::add_bulk(*request_, to_string(cmd));
124  resp3::add_bulk(*request_, key);
125 
126  for (; begin != end; ++begin)
127  resp3::add_bulk(*request_, *begin);
128  }
