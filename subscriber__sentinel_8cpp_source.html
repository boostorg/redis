29// Connects to a Redis instance over sentinel and performs failover in
30// case of disconnection, see
31// https://redis.io/docs/reference/sentinel-clients. This example
32// assumes a sentinel and a redis server running on localhost.
33
34net::awaitable<void> receive_pushes(std::shared_ptr<connection> db)
35{
36 for (std::vector<node<std::string>> resp;;) {
37 co_await db->async_receive_push(adapt(resp));
38 print_push(resp);
39 resp.clear();
40 }
41}

43net::awaitable<endpoint> resolve()
44{
45 // A list of sentinel addresses from which only one is responsive
46 // to simulate sentinels that are down.
47 std::vector<endpoint> const endpoints
48 { {"foo", "26379"}
49 , {"bar", "26379"}
50 , {"127.0.0.1", "26379"}
51 };
52
53 request req1;
54 req1.push("SENTINEL", "get-master-addr-by-name", "mymaster");
55 req1.push("QUIT");
56
57 auto ex = co_await net::this_coro::executor;
58 connection conn{ex};
59
60 std::tuple<std::optional<std::array<std::string, 2>>, aedis::ignore> addr;
61 for (auto ep : endpoints) {
62 boost::system::error_code ec;
63 co_await conn.async_run(ep, req1, adapt(addr), {}, net::redirect_error(net::use_awaitable, ec));
64 conn.reset_stream();
65 std::cout << ec.message() << std::endl;
66 if (std::get<0>(addr))
67 break;
68 }
69
70 endpoint ep;
71 if (std::get<0>(addr)) {
72 ep.host = std::get<0>(addr).value().at(0);
73 ep.port = std::get<0>(addr).value().at(1);
74 }
75
76 co_return ep;
77}

79net::awaitable<void> reconnect(std::shared_ptr<connection> db)
80{
81 request req2;
82 req2.push("SUBSCRIBE", "channel");
83
84 auto ex = co_await net::this_coro::executor;
85 stimer timer{ex};
86 for (;;) {
87 auto ep = co_await net::co_spawn(ex, resolve(), net::use_awaitable);
88 if (!aedis::is_valid(ep)) {
89 std::clog << "Can't resolve master name" << std::endl;
90 co_return;
91 }
92
93 boost::system::error_code ec;
94 co_await db->async_run(ep, req2, adapt(), {}, net::redirect_error(net::use_awaitable, ec));
95 std::cout << ec.message() << std::endl;
96 std::cout << "Starting the failover." << std::endl;
97 timer.expires_after(std::chrono::seconds{1});
98 co_await timer.async_wait();
99 }
100}

102int main()
103{
104 try {
105 net::io_context ioc;
106 auto db = std::make_shared<connection>(ioc);
107 net::co_spawn(ioc, receive_pushes(db), net::detached);
108 net::co_spawn(ioc, reconnect(db), net::detached);
109 net::signal_set signals(ioc, SIGINT, SIGTERM);
110 signals.async_wait([&](auto, auto){ ioc.stop(); });
111 ioc.run();
112 } catch (std::exception const& e) {
113 std::cerr << "Error: " << e.what() << std::endl;
114 }
115}
