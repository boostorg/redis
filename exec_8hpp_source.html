27template <
28 class AsyncStream,
29 class Adapter,
30 class DynamicBuffer
31 >
32struct exec_op {
33 AsyncStream* socket = nullptr;
34 request const* req = nullptr;
35 Adapter adapter;
36 DynamicBuffer dbuf{};
37 std::size_t n_cmds = 0;
38 std::size_t size = 0;
39 boost::asio::coroutine coro{};
40
41 template <class Self>
42 void operator()( Self& self
43 , boost::system::error_code ec = {}
44 , std::size_t n = 0)
45 {
46 reenter (coro) for (;;)
47 {
48 if (req) {
49 yield
50 boost::asio::async_write(
51 *socket,
52 boost::asio::buffer(req->payload()),
53 std::move(self));
54
55 if (ec || n_cmds == 0) {
56 self.complete(ec, n);
57 return;
58 }
59
60 req = nullptr;
61 }
62
63 yield resp3::async_read(*socket, dbuf, adapter, std::move(self));
64 if (ec) {
65 self.complete(ec, 0);
66 return;
67 }
68
69 size += n;
70 if (--n_cmds == 0) {
71 self.complete(ec, size);
72 return;
73 }
74 }
75 }
76};

98template <
99 class AsyncStream,
100 class Timer,
101 class Adapter,
102 class DynamicBuffer
103 >
104struct exec_with_timeout_op {
105 AsyncStream* socket = nullptr;
106 Timer* timer = nullptr;
107 request const* req = nullptr;
108 Adapter adapter;
109 DynamicBuffer dbuf{};
110 boost::asio::coroutine coro{};
111
112 template <class Self>
113 void operator()( Self& self
114 , std::array<std::size_t, 2> order = {}
115 , boost::system::error_code ec1 = {}
116 , std::size_t n = 0
117 , boost::system::error_code ec2 = {})
118 {
119 reenter (coro)
120 {
121 yield
122 boost::asio::experimental::make_parallel_group(
123 [this](auto token) { return detail::async_exec(*socket, *req, adapter, dbuf, token);},
124 [this](auto token) { return timer->async_wait(token);}
125 ).async_wait(
126 boost::asio::experimental::wait_for_one(),
127 std::move(self));
128
129 switch (order[0]) {
130 case 0: self.complete(ec1, n); break;
131 case 1:
132 {
133 if (ec2) {
134 self.complete(ec2, 0);
135 } else {
136 self.complete(aedis::error::exec_timeout, 0);
137 }
138
139 } break;
140
141 default: BOOST_ASSERT(false);
142 }
143 }
144 }
145};
