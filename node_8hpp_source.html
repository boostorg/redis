46template <class String>
47auto to_string(node<String> const& in)
48{
49 std::string out;
50 out += std::to_string(in.depth);
51 out += '\t';
52 out += to_string(in.data_type);
53 out += '\t';
54 out += std::to_string(in.aggregate_size);
55 out += '\t';
56 if (!is_aggregate(in.data_type))
57 out.append(in.value.data(), in.value.size());
58
59 return out;
60}

68template <class String>
69auto operator==(node<String> const& a, node<String> const& b)
70{
71 return a.aggregate_size == b.aggregate_size
72 && a.depth == b.depth
73 && a.data_type == b.data_type
74 && a.value == b.value;
75};
