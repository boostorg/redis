47template <class Request>
48void to_bulk(Request& to, boost::string_view data)
49{
50 auto const str = std::to_string(data.size());
51
52 to += to_code(type::blob_string);
53 to.append(std::cbegin(str), std::cend(str));
54 to += separator;
55 to.append(std::cbegin(data), std::cend(data));
56 to += separator;
57}
58
59template <class Request, class T, typename = typename std::enable_if<std::is_integral<T>::value>::type>
60void to_bulk(Request& to, T n)
61{
62 auto const s = std::to_string(n);
63 to_bulk(to, boost::string_view{s});
64}

91template <class ...Ts>
92struct add_bulk_impl<boost::hana::tuple<Ts...>> {
93 template <class Request>
94 static void add(Request& to, boost::hana::tuple<Ts...> const& from)
95 {
96 using boost::hana::for_each;
97
98 // Fold expressions is C++17 so we use hana.
99 //(detail::add_bulk(*request_, args), ...);
100
101 for_each(from, [&](auto const& e) {
102 using namespace aedis::resp3;
103 to_bulk(to, e);
104 });
105 }
106};

108template <class Request>
109void add_header(Request& to, type t, std::size_t size)
110{
111 auto const str = std::to_string(size);
112
113 to += to_code(t);
114 to.append(std::cbegin(str), std::cend(str));
115 to += separator;
116}

118template <class Request, class T>
119void add_bulk(Request& to, T const& data)
120{

231 template <class... Ts>
232 void push(boost::string_view cmd, Ts const&... args)
233 {
234 using boost::hana::for_each;
235 using boost::hana::make_tuple;
236 using resp3::type;
237
238 auto constexpr pack_size = sizeof...(Ts);
239 detail::add_header(payload_, type::array, 1 + pack_size);
240 detail::add_bulk(payload_, cmd);
241 detail::add_bulk(payload_, make_tuple(args...));
242
243 if (!detail::has_push_response(cmd))
244 ++commands_;
245 }

268 template <class Key, class ForwardIterator>
269 void push_range2(boost::string_view cmd, Key const& key, ForwardIterator begin, ForwardIterator end)
270 {
271 using value_type = typename std::iterator_traits<ForwardIterator>::value_type;
272 using resp3::type;
273
274 if (begin == end)
275 return;
276
277 auto constexpr size = detail::bulk_counter<value_type>::size;
278 auto const distance = std::distance(begin, end);
279 detail::add_header(payload_, type::array, 2 + size * distance);
280 detail::add_bulk(payload_, cmd);
281 detail::add_bulk(payload_, key);
282
283 for (; begin != end; ++begin)
284 detail::add_bulk(payload_, *begin);
285
286 if (!detail::has_push_response(cmd))
287 ++commands_;
288 }
289
307 template <class ForwardIterator>
308 void push_range2(boost::string_view cmd, ForwardIterator begin, ForwardIterator end)
309 {
310 using value_type = typename std::iterator_traits<ForwardIterator>::value_type;
311 using resp3::type;
312
313 if (begin == end)
314 return;
315
316 auto constexpr size = detail::bulk_counter<value_type>::size;
317 auto const distance = std::distance(begin, end);
318 detail::add_header(payload_, type::array, 1 + size * distance);
319 detail::add_bulk(payload_, cmd);
320
321 for (; begin != end; ++begin)
322 detail::add_bulk(payload_, *begin);
323
324 if (!detail::has_push_response(cmd))
325 ++commands_;
326 }

336 template <class Key, class Range>
337 void push_range(boost::string_view cmd, Key const& key, Range const& range)

339 using std::begin;
