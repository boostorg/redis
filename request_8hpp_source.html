74  template <class... Ts>
75  void push(boost::string_view cmd, Ts const&... args)
76  {
77  using boost::hana::for_each;
78  using boost::hana::make_tuple;
79  using resp3::type;
80 
81  auto const before = payload_.size();
82  auto constexpr pack_size = sizeof...(Ts);
83  resp3::add_header(payload_, type::array, 1 + pack_size);
84  resp3::add_bulk(payload_, cmd);
85  resp3::add_bulk(payload_, make_tuple(args...));
86 
87  auto const after = payload_.size();
88  if (!has_push_response(cmd))
89  ++commands_;
90  }

113  template <class Key, class ForwardIterator>
114  void push_range2(boost::string_view cmd, Key const& key, ForwardIterator begin, ForwardIterator end)
115  {
116  using value_type = typename std::iterator_traits<ForwardIterator>::value_type;
117  using resp3::type;
118 
119  if (begin == end)
120  return;
121 
122  auto const before = payload_.size();
123 
124  auto constexpr size = resp3::bulk_counter<value_type>::size;
125  auto const distance = std::distance(begin, end);
126  resp3::add_header(payload_, type::array, 2 + size * distance);
127  resp3::add_bulk(payload_, cmd);
128  resp3::add_bulk(payload_, key);
129 
130  for (; begin != end; ++begin)
131  resp3::add_bulk(payload_, *begin);
132 
133  auto const after = payload_.size();
134  if (!has_push_response(cmd))
135  ++commands_;
136  }
