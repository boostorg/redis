45template <class Request>
46void to_bulk(Request& to, std::string_view data)
47{
48 auto const str = std::to_string(data.size());
49
50 to += to_code(type::blob_string);
51 to.append(std::cbegin(str), std::cend(str));
52 to += separator;
53 to.append(std::cbegin(data), std::cend(data));
54 to += separator;
55}

57template <class Request, class T, typename = typename std::enable_if<std::is_integral<T>::value>::type>
58void to_bulk(Request& to, T n)
59{
60 auto const s = std::to_string(n);
61 to_bulk(to, std::string_view{s});
62}

104template <class Request>
105void add_header(Request& to, type t, std::size_t size)

260 template <class... Ts>
261 void push(std::string_view cmd, Ts const&... args)
262 {
263 using resp3::type;
264
265 auto constexpr pack_size = sizeof...(Ts);
266 detail::add_header(payload_, type::array, 1 + pack_size);
267 detail::add_bulk(payload_, cmd);
268 detail::add_bulk(payload_, std::tie(std::forward<Ts const&>(args)...));
269
270 check_cmd(cmd);
271 }

294 template <class Key, class ForwardIterator>
295 void push_range(std::string_view cmd, Key const& key, ForwardIterator begin, ForwardIterator end,
296 typename std::iterator_traits<ForwardIterator>::value_type * = nullptr)
297 {
298 using value_type = typename std::iterator_traits<ForwardIterator>::value_type;
299 using resp3::type;
300
301 if (begin == end)
302 return;
303
304 auto constexpr size = detail::bulk_counter<value_type>::size;
305 auto const distance = std::distance(begin, end);
306 detail::add_header(payload_, type::array, 2 + size * distance);
307 detail::add_bulk(payload_, cmd);
308 detail::add_bulk(payload_, key);
309
310 for (; begin != end; ++begin)
311 detail::add_bulk(payload_, *begin);
312
313 check_cmd(cmd);
314 }
315
