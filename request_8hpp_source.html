48template <class Request>
49void to_bulk(Request& to, boost::string_view data)
50{
51 auto const str = std::to_string(data.size());
52
53 to += to_code(type::blob_string);
54 to.append(std::cbegin(str), std::cend(str));
55 to += separator;
56 to.append(std::cbegin(data), std::cend(data));
57 to += separator;
58}
59
60template <class Request, class T, typename = typename std::enable_if<std::is_integral<T>::value>::type>
61void to_bulk(Request& to, T n)
62{
63 auto const s = std::to_string(n);
64 to_bulk(to, boost::string_view{s});
65}

81template <class U, class V>
82struct add_bulk_impl<std::pair<U, V>> {
83 template <class Request>
84 static void add(Request& to, std::pair<U, V> const& from)
85 {
86 using namespace aedis::resp3;
87 to_bulk(to, from.first);

92template <class ...Ts>
93struct add_bulk_impl<boost::hana::tuple<Ts...>> {
94 template <class Request>
95 static void add(Request& to, boost::hana::tuple<Ts...> const& from)
96 {
97 using boost::hana::for_each;
98
99 // Fold expressions is C++17 so we use hana.
100 //(detail::add_bulk(*request_, args), ...);
101
102 for_each(from, [&](auto const& e) {
103 using namespace aedis::resp3;
104 to_bulk(to, e);
105 });
106 }
107};

109template <class Request>
110void add_header(Request& to, type t, std::size_t size)
111{
112 auto const str = std::to_string(size);
113
114 to += to_code(t);
115 to.append(std::cbegin(str), std::cend(str));
116 to += separator;
117}

119template <class Request, class T>
120void add_bulk(Request& to, T const& data)
121{
122 detail::add_bulk_impl<T>::add(to, data);
123}

245 template <class... Ts>
246 void push(boost::string_view cmd, Ts const&... args)
247 {
248 using boost::hana::for_each;
249 using boost::hana::make_tuple;
250 using resp3::type;
251
252 auto constexpr pack_size = sizeof...(Ts);
253 detail::add_header(payload_, type::array, 1 + pack_size);
254 detail::add_bulk(payload_, cmd);
255 detail::add_bulk(payload_, make_tuple(args...));
256
257 if (!detail::has_push_response(cmd))
258 ++commands_;
259 }
260
282 template <class Key, class ForwardIterator>
283 void push_range(boost::string_view cmd, Key const& key, ForwardIterator begin, ForwardIterator end,
284 typename std::iterator_traits<ForwardIterator>::value_type * = nullptr)
285 {
286 using value_type = typename std::iterator_traits<ForwardIterator>::value_type;
287 using resp3::type;
288
289 if (begin == end)
290 return;
291
292 auto constexpr size = detail::bulk_counter<value_type>::size;
293 auto const distance = std::distance(begin, end);
294 detail::add_header(payload_, type::array, 2 + size * distance);
295 detail::add_bulk(payload_, cmd);
296 detail::add_bulk(payload_, key);
297
298 for (; begin != end; ++begin)
299 detail::add_bulk(payload_, *begin);
300
301 if (!detail::has_push_response(cmd))
302 ++commands_;
303 }
304
322 template <class ForwardIterator>
323 void push_range(boost::string_view cmd, ForwardIterator begin, ForwardIterator end,
324 typename std::iterator_traits<ForwardIterator>::value_type * = nullptr)
325 {
326 using value_type = typename std::iterator_traits<ForwardIterator>::value_type;
327 using resp3::type;
328
329 if (begin == end)
330 return;
331
332 auto constexpr size = detail::bulk_counter<value_type>::size;
333 auto const distance = std::distance(begin, end);
334 detail::add_header(payload_, type::array, 1 + size * distance);
335 detail::add_bulk(payload_, cmd);
336
337 for (; begin != end; ++begin)
338 detail::add_bulk(payload_, *begin);
339
340 if (!detail::has_push_response(cmd))
341 ++commands_;
342 }

353 void push_range(boost::string_view cmd, Key const& key, Range const& range,

356 using std::begin;
