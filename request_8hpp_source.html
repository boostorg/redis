48template <class Request>
49void to_bulk(Request& to, boost::string_view data)
50{
51 auto const str = std::to_string(data.size());
52
53 to += to_code(type::blob_string);
54 to.append(std::cbegin(str), std::cend(str));
55 to += separator;
56 to.append(std::cbegin(data), std::cend(data));
57 to += separator;
58}
59
60template <class Request, class T, typename = typename std::enable_if<std::is_integral<T>::value>::type>
61void to_bulk(Request& to, T n)
62{
63 auto const s = std::to_string(n);
64 to_bulk(to, boost::string_view{s});
65}

83template <class U, class V>

94template <class ...Ts>
95struct add_bulk_impl<boost::hana::tuple<Ts...>> {
96 template <class Request>
97 static void add(Request& to, boost::hana::tuple<Ts...> const& from)
98 {
99 using boost::hana::for_each;
100
101 // Fold expressions is C++17 so we use hana.
102 //(detail::add_bulk(*request_, args), ...);
103
104 for_each(from, [&](auto const& e) {
105 using namespace aedis::resp3;
106 to_bulk(to, e);
107 });
108 }
109};

111template <class Request>
112void add_header(Request& to, type t, std::size_t size)
113{
114 auto const str = std::to_string(size);
115
116 to += to_code(t);
117 to.append(std::cbegin(str), std::cend(str));
118 to += separator;
119}

121template <class Request, class T>
122void add_bulk(Request& to, T const& data)
123{
124 detail::add_bulk_impl<T>::add(to, data);
125}

264 template <class... Ts>
265 void push(boost::string_view cmd, Ts const&... args)
266 {
267 using boost::hana::for_each;
268 using boost::hana::make_tuple;
269 using resp3::type;
270
271 auto constexpr pack_size = sizeof...(Ts);
272 detail::add_header(payload_, type::array, 1 + pack_size);
273 detail::add_bulk(payload_, cmd);
274 detail::add_bulk(payload_, make_tuple(args...));
275
276 check_cmd(cmd);
277 }

300 template <class Key, class ForwardIterator>
301 void push_range(boost::string_view cmd, Key const& key, ForwardIterator begin, ForwardIterator end,
302 typename std::iterator_traits<ForwardIterator>::value_type * = nullptr)
303 {
304 using value_type = typename std::iterator_traits<ForwardIterator>::value_type;
305 using resp3::type;
306
307 if (begin == end)
308 return;
309
310 auto constexpr size = detail::bulk_counter<value_type>::size;
311 auto const distance = std::distance(begin, end);
312 detail::add_header(payload_, type::array, 2 + size * distance);
313 detail::add_bulk(payload_, cmd);
314 detail::add_bulk(payload_, key);
315
316 for (; begin != end; ++begin)
317 detail::add_bulk(payload_, *begin);
318
319 check_cmd(cmd);
320 }
321
339 template <class ForwardIterator>
340 void push_range(boost::string_view cmd, ForwardIterator begin, ForwardIterator end,
341 typename std::iterator_traits<ForwardIterator>::value_type * = nullptr)
342 {
343 using value_type = typename std::iterator_traits<ForwardIterator>::value_type;
344 using resp3::type;
345
346 if (begin == end)
347 return;
348
349 auto constexpr size = detail::bulk_counter<value_type>::size;
350 auto const distance = std::distance(begin, end);
351 detail::add_header(payload_, type::array, 1 + size * distance);
352 detail::add_bulk(payload_, cmd);
353
354 for (; begin != end; ++begin)
355 detail::add_bulk(payload_, *begin);
356
357 check_cmd(cmd);
358 }

393private:
394 void check_cmd(boost::string_view cmd)
395 {
396 if (!detail::has_push_response(cmd))
397 ++commands_;
398
399 has_hello_priority_ = detail::is_hello(cmd) && cfg_.hello_with_priority;
400 }
401
402 config cfg_;
403 std::pmr::string payload_;
404 std::size_t commands_ = 0;
405 bool has_hello_priority_ = false;
