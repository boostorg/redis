29 template <
30  class AsyncReadStream,
31  class DynamicBuffer,
32  class ResponseAdapter>
33 class parse_op {
34 private:
35  AsyncReadStream& stream_;
36  DynamicBuffer buf_;
37  parser<ResponseAdapter> parser_;
38  std::size_t consumed_;
39  std::size_t buffer_size_;
40  boost::asio::coroutine coro_{};
41 
42 public:
43  parse_op(AsyncReadStream& stream, DynamicBuffer buf, ResponseAdapter adapter)
44  : stream_ {stream}
45  , buf_ {buf}
46  , parser_ {adapter}
47  , consumed_{0}
48  { }
49 
50  template <class Self>
51  void operator()( Self& self
52  , boost::system::error_code ec = {}
53  , std::size_t n = 0)
54  {
55  reenter (coro_) for (;;) {
56  if (parser_.bulk() == type::invalid) {
57  yield
58  boost::asio::async_read_until(stream_, buf_, "\r\n", std::move(self));
59 
60  if (ec) {
61  self.complete(ec, 0);
62  return;
63  }
64  } else {
65  // On a bulk read we can't read until delimiter since the
66  // payload may contain the delimiter itself so we have to
67  // read the whole chunk. However if the bulk blob is small
68  // enough it may be already on the buffer (from the last
69  // read), in which case there is no need of initiating
70  // another async op, otherwise we have to read the missing
71  // bytes.
72  if (buf_.size() < (parser_.bulk_length() + 2)) {
73  buffer_size_ = buf_.size();
74  buf_.grow(parser_.bulk_length() + 2 - buffer_size_);
75 
76  yield
77  boost::asio::async_read(
78  stream_,
79  buf_.data(buffer_size_, parser_.bulk_length() + 2 - buffer_size_),
80  boost::asio::transfer_all(),
81  std::move(self));
82 
83  if (ec) {
84  self.complete(ec, 0);
85  return;
86  }
87  }
88 
89  n = parser_.bulk_length() + 2;
90  BOOST_ASSERT(buf_.size() >= n);
91  }
92 
93  n = parser_.consume((char const*)buf_.data(0, n).data(), n, ec);
94  if (ec) {
95  self.complete(ec, 0);
96  return;
97  }
98 
99  buf_.consume(n);
100  consumed_ += n;
101  if (parser_.done()) {
102  self.complete({}, consumed_);
103  return;
104  }
105  }
106  }
107 };
