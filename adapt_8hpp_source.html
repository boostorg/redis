61template <class Tuple>
62class static_adapter {
63private:
64 static constexpr auto size = std::tuple_size<Tuple>::value;
65 using adapter_tuple = boost::mp11::mp_transform<adapter::adapter_t, Tuple>;
66 using variant_type = boost::mp11::mp_rename<adapter_tuple, std::variant>;
67 using adapters_array_type = std::array<variant_type, size>;
68
69 adapters_array_type adapters_;
70 std::size_t max_read_size_;
71
72public:
73 explicit static_adapter(Tuple& r, std::size_t max_read_size)
74 : max_read_size_{max_read_size}
75 {
76 adapter::detail::assigner<size - 1>::assign(adapters_, r);
77 }
78
79 [[nodiscard]]
80 auto get_supported_response_size() const noexcept
81 { return size;}
82
83 [[nodiscard]]
84 auto get_max_read_size(std::size_t) const noexcept
85 { return max_read_size_;}
86
87 void
88 operator()(
89 std::size_t i,
90 resp3::node<std::string_view> const& nd,
91 boost::system::error_code& ec)
92 {
93 using std::visit;
94 // I am usure whether this should be an error or an assertion.
95 BOOST_ASSERT(i < adapters_.size());
96 visit([&](auto& arg){arg(nd, ec);}, adapters_.at(i));
97 }
98};

162template <class Adapter>
163class wrapper {
164public:
165 explicit wrapper(Adapter adapter) : adapter_{adapter} {}
166
167 void operator()(resp3::node<std::string_view> const& node, boost::system::error_code& ec)
168 { return adapter_(0, node, ec); }
169
170 [[nodiscard]]
171 auto get_supported_response_size() const noexcept
172 { return adapter_.get_supported_response_size();}
173
174 [[nodiscard]]
175 auto get_max_read_size(std::size_t) const noexcept
176 { return adapter_.get_max_read_size(0); }
177
178private:
179 Adapter adapter_;
180};
