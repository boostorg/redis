32 template <class Storage>
33 void add_header(Storage& to, int size)
34 {
35  // std::string does not support allocators.
36  using std::to_string;
37  auto const str = to_string(size);
38 
39  to += "*";
40  to.append(std::cbegin(str), std::cend(str));
41  to += "\r\n";
42 }
43 
44 template <class Storage>
45 void add_bulk(Storage& to, std::string_view data)
46 {
47  // std::string does not support allocators.
48  using std::to_string;
49  auto const str = to_string(std::size(data));
50 
51  to += "$";
52  to.append(std::cbegin(str), std::cend(str));
53  to += "\r\n";
54  to += data;
55  to += "\r\n";
56 }
57 
58 template <class Storage, class T>
59 void add_bulk(Storage& to, T const& data, typename std::enable_if<needs_to_string<T>::value, bool>::type = false)
60 {
61  using std::to_string;
62  auto const s = to_string(data);
63  add_bulk(to, s);
64 }

66 // Overload for pairs.
67 // TODO: Overload for tuples.

69 void add_bulk(Storage& to, std::pair<T1, T2> const& pair)
70 {
71  add_bulk(to, pair.first);
72  add_bulk(to, pair.second);
73 }
