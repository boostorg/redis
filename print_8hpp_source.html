21void print_aggr(std::vector<aedis::resp3::node<std::string>>& v)
22{
23 if (std::empty(v))
24 return;
25
26 auto const m = aedis::resp3::element_multiplicity(v.front().data_type);
27 for (auto i = 0lu; i < m * v.front().aggregate_size; ++i)
28 std::cout << v[i + 1].value << " ";
29 std::cout << "\n";
30 v.clear();
31}
