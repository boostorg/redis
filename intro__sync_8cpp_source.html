16namespace net = boost::asio;
17using aedis::adapt;
18using aedis::resp3::request;
19using connection = aedis::connection;
20
21template <class Adapter>
22auto exec(std::shared_ptr<connection> conn, request const& req, Adapter adapter)
23{
24 net::dispatch(
25 conn->get_executor(),
26 net::deferred([&]() { return conn->async_exec(req, adapter, net::deferred); }))
27 (net::use_future).get();
28}
29
30auto logger = [](auto const& ec)
31 { std::clog << "Run: " << ec.message() << std::endl; };
32
33int main()
34{
35 try {
36 net::io_context ioc{1};
37
38 auto conn = std::make_shared<connection>(ioc);
39 net::ip::tcp::resolver resv{ioc};
40 auto const res = resv.resolve("127.0.0.1", "6379");
41 net::connect(conn->next_layer(), res);
42 std::thread t{[conn, &ioc]() {
43 conn->async_run(logger);
44 ioc.run();
45 }};
46
47 request req;
48 req.get_config().cancel_on_connection_lost = true;
49 req.push("HELLO", 3);
50 req.push("PING");
51 req.push("QUIT");
52
53 std::tuple<aedis::ignore, std::string, aedis::ignore> resp;
54 exec(conn, req, adapt(resp));
55
56 std::cout << "Response: " << std::get<1>(resp) << std::endl;
57
58 t.join();
59 } catch (std::exception const& e) {
60 std::cerr << e.what() << std::endl;
61 }
62}
