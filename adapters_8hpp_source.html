66 void set_on_resp3_error(resp3::type t, boost::system::error_code& ec)
67 {
68  switch (t) {
69  case resp3::type::simple_error: ec = adapter::error::simple_error; return;
70  case resp3::type::blob_error: ec = adapter::error::blob_error; return;
71  case resp3::type::null: ec = adapter::error::null; return;
72  default: return;
73  }

130 template <class Result>
131 class set_impl {
132 private:
133  typename Result::iterator hint_;
134 
135 public:
136  void on_value_available(Result& result)
137  { hint_ = std::end(result); }
138 
139  void
140  operator()(
141  Result& result,
142  resp3::node<boost::string_view> const& nd,
143  boost::system::error_code& ec)
144  {
145  set_on_resp3_error(nd.data_type, ec);
146  if (ec)
147  return;
148 
149  if (is_aggregate(nd.data_type)) {
150  if (nd.data_type != resp3::type::set)
151  ec = error::expects_set_aggregate;
152  return;
153  }
154 
155  assert(nd.aggregate_size == 1);
156 
157  if (nd.depth < 1) {
158  ec = adapter::error::expects_set_aggregate;
159  return;
160  }
161 
162  typename Result::key_type obj;
163  from_string(obj, nd.value, ec);
164  hint_ = result.insert(hint_, std::move(obj));
165  }
166 };

168 template <class Result>
169 class map_impl {
170 private:
171  typename Result::iterator current_;
172  bool on_key_ = true;
173 
174 public:
175  void on_value_available(Result& result)
176  { current_ = std::end(result); }
177 
178  void
179  operator()(
180  Result& result,
181  resp3::node<boost::string_view> const& nd,
182  boost::system::error_code& ec)
183  {
184  set_on_resp3_error(nd.data_type, ec);
185  if (ec)
186  return;
187 
188  if (is_aggregate(nd.data_type)) {
189  if (element_multiplicity(nd.data_type) != 2)
190  ec = error::expects_map_like_aggregate;
191  return;
192  }
193 
194  assert(nd.aggregate_size == 1);
195 
196  if (nd.depth < 1) {
197  ec = adapter::error::expects_map_like_aggregate;
198  return;
199  }
200 
201  if (on_key_) {
202  typename Result::key_type obj;
203  from_string(obj, nd.value, ec);
204  current_ = result.insert(current_, {std::move(obj), {}});
205  } else {
206  typename Result::mapped_type obj;
207  from_string(obj, nd.value, ec);
208  current_->second = std::move(obj);
209  }
210 
211  on_key_ = !on_key_;
212  }
213 };

215 template <class Result>
216 class vector_impl {
217 public:
218  void on_value_available(Result& ) { }
219 
220  void
221  operator()(
222  Result& result,
223  resp3::node<boost::string_view> const& nd,
224  boost::system::error_code& ec)
225  {
226  set_on_resp3_error(nd.data_type, ec);
227  if (ec)
228  return;
229 
230  if (is_aggregate(nd.data_type)) {
231  auto const m = element_multiplicity(nd.data_type);
232  result.reserve(result.size() + m * nd.aggregate_size);
233  } else {
234  result.push_back({});
235  from_string(result.back(), nd.value, ec);
236  }
237  }

240 template <class Result>
241 class array_impl {
242 private:
243  int i_ = -1;
244 
245 public:
246  void on_value_available(Result& ) { }
247 
248  void
249  operator()(
250  Result& result,
251  resp3::node<boost::string_view> const& nd,
252  boost::system::error_code& ec)
253  {
254  set_on_resp3_error(nd.data_type, ec);
255  if (ec)
256  return;
257 
258  if (is_aggregate(nd.data_type)) {
259  if (i_ != -1) {
260  ec = adapter::error::nested_aggregate_unsupported;
261  return;
262  }
263 
264  if (result.size() != nd.aggregate_size * element_multiplicity(nd.data_type)) {
265  ec = error::incompatible_size;
266  return;
267  }
268  } else {
269  if (i_ == -1) {
270  ec = adapter::error::expects_aggregate;
271  return;
272  }
273 
274  assert(nd.aggregate_size == 1);
275  from_string(result.at(i_), nd.value, ec);
276  }
277 
278  ++i_;
279  }
280 };

380  wrapper(boost::optional<T>* o = nullptr) : result_(o), impl_{} {}

383  operator()(

385  boost::system::error_code& ec)
386  {
