277  void cancel_run()
278  {
279  socket_->close();
280  read_timer_.cancel();
281  check_idle_timer_.cancel();
282  writer_timer_.cancel();
283  ping_timer_.cancel();
284 
285  push_channel_.cancel();
286 
287  // Cancel own pings if there is any waiting.
288  auto point = std::stable_partition(std::begin(reqs_), std::end(reqs_), [](auto const& ptr) {
289  return !ptr->req->is_internal();
290  });
291 
292  std::for_each(point, std::end(reqs_), [](auto const& ptr) {
293  ptr->stop = true;
294  ptr->timer.cancel();
295  });
296 
297  reqs_.erase(point, std::end(reqs_));
298  }

428  void coalesce_requests()
429  {
430  // Coaleces all requests: Copies the request to the variables
431  // that won't be touched while async_write is suspended.
432  BOOST_ASSERT(write_buffer_.empty());
433  BOOST_ASSERT(!reqs_.empty());
434 
435  auto const size = cfg_.coalesce_requests ? reqs_.size() : 1;
436  for (auto i = 0UL; i < size; ++i) {
437  write_buffer_ += reqs_.at(i)->req->payload();
438  cmds_ += reqs_.at(i)->req->commands();
439  reqs_.at(i)->written = true;
440  }
441  }
