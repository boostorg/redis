317  void cancel_run()
318  {
319  if (socket_)
320  socket_->close();
321 
322  read_timer_.cancel();
323  check_idle_timer_.cancel();
324  writer_timer_.cancel();
325  ping_timer_.cancel();
326 
327  // Cancel own pings if there is any waiting.
328  auto point = std::stable_partition(std::begin(reqs_), std::end(reqs_), [](auto const& ptr) {
329  return !ptr->req->close_on_run_completion;
330  });
331 
332  std::for_each(point, std::end(reqs_), [](auto const& ptr) {
333  ptr->stop = true;
334  ptr->timer.cancel();
335  });
336 
337  reqs_.erase(point, std::end(reqs_));
338  }

486  void coalesce_requests()
487  {
488  // Coaleces all requests: Copies the request to the variables
489  // that won't be touched while async_write is suspended.
490  BOOST_ASSERT(write_buffer_.empty());
491  BOOST_ASSERT(!reqs_.empty());
492 
493  auto const size = cfg_.coalesce_requests ? reqs_.size() : 1;
494  for (auto i = 0UL; i < size; ++i) {
495  write_buffer_ += reqs_.at(i)->req->payload();
496  cmds_ += reqs_.at(i)->req->size();
497  reqs_.at(i)->written = true;
498  }
499  }

527 template <class T>
528 char const* to_string(typename connection<T>::event e)
529 {
530  using event_type = typename connection<T>::event;
531  switch (e) {
532  case event_type::resolve: return "resolve";
533  case event_type::connect: return "connect";
534  case event_type::hello: return "hello";

537  default: BOOST_ASSERT_MSG(false, "to_string: unhandled event.");
