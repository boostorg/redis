171  std::size_t cancel(operation op)
172  {
173  switch (op) {
174  case operation::exec:
175  {
176  for (auto& e: reqs_) {
177  e->stop = true;
178  e->timer.cancel_one();
179  }
180 
181  auto const ret = reqs_.size();
182  reqs_ = {};
183  return ret;
184  }
185  case operation::run:
186  {
187  if (socket_)
188  socket_->close();
189 
190  read_timer_.cancel();
191  check_idle_timer_.cancel();
192  writer_timer_.cancel();
193  ping_timer_.cancel();
194 
195  auto point = std::stable_partition(std::begin(reqs_), std::end(reqs_), [](auto const& ptr) {
196  return !ptr->req->close_on_run_completion;
197  });
198 
199  // Cancel own pings if there are any waiting.
200  std::for_each(point, std::end(reqs_), [](auto const& ptr) {
201  ptr->stop = true;
202  ptr->timer.cancel();
203  });
204 
205  reqs_.erase(point, std::end(reqs_));
206  return 1U;
207  }
208  case operation::receive_event:
209  {
210  event_channel_.cancel();
211  return 1U;
212  }
213  case operation::receive_push:
214  {
215  push_channel_.cancel();
216  return 1U;
217  }
218  }
219 
220  return 0;
221  }

547  void coalesce_requests()
548  {
549  // Coaleces all requests: Copies the request to the variables
550  // that won't be touched while async_write is suspended.
551  BOOST_ASSERT(write_buffer_.empty());
552  BOOST_ASSERT(!reqs_.empty());
553 
554  auto const size = cfg_.coalesce_requests ? reqs_.size() : 1;
555  for (auto i = 0UL; i < size; ++i) {
556  write_buffer_ += reqs_.at(i)->req->payload();
557  cmds_ += reqs_.at(i)->req->size();
558  reqs_.at(i)->written = true;
559  }
560  }
