20template <class Stream>
21struct handshake_op {
22 Stream* stream;
23 aedis::detail::conn_timer_t<typename Stream::executor_type>* timer;
24 boost::asio::coroutine coro{};
25
26 template <class Self>
27 void operator()( Self& self
28 , std::array<std::size_t, 2> order = {}
29 , boost::system::error_code ec1 = {}
30 , boost::system::error_code ec2 = {})
31 {
32 reenter (coro)
33 {
34 yield
35 boost::asio::experimental::make_parallel_group(
36 [this](auto token)
37 {
38 return stream->async_handshake(boost::asio::ssl::stream_base::client, token);
39 },
40 [this](auto token) { return timer->async_wait(token);}
41 ).async_wait(
42 boost::asio::experimental::wait_for_one(),
43 std::move(self));
44
45 if (is_cancelled(self)) {
46 self.complete(boost::asio::error::operation_aborted);
47 return;
48 }
49
50 switch (order[0]) {
51 case 0: self.complete(ec1); return;
52 case 1:
53 {
54 BOOST_ASSERT_MSG(!ec2, "handshake_op: Incompatible state.");
55 self.complete(error::ssl_handshake_timeout);
56 return;
57 }
58
59 default: BOOST_ASSERT(false);
60 }
61 }
62 }
63};
64
65template <
66 class Stream,
67 class CompletionToken
68 >
69auto async_handshake(
70 Stream& stream,
71 aedis::detail::conn_timer_t<typename Stream::executor_type>& timer,
72 CompletionToken&& token)
73{
74 return boost::asio::async_compose
75 < CompletionToken
76 , void(boost::system::error_code)
77 >(handshake_op<Stream>{&stream, &timer}, token, stream, timer);
78}

80template <class Conn, class Timer>
81struct ssl_connect_with_timeout_op {
82 Conn* conn = nullptr;
83 boost::asio::ip::tcp::resolver::results_type const* endpoints = nullptr;
84 typename Conn::timeouts ts;
85 Timer* timer = nullptr;
86 boost::asio::coroutine coro{};
87
88 template <class Self>
89 void operator()( Self& self
90 , boost::system::error_code ec = {}
91 , boost::asio::ip::tcp::endpoint const& = {})
92 {
93 reenter (coro)
94 {
95 timer->expires_after(ts.connect_timeout);
96 yield
97 aedis::detail::async_connect(
98 conn->lowest_layer(), *timer, *endpoints, std::move(self));
99 AEDIS_CHECK_OP0();
100
101 timer->expires_after(ts.handshake_timeout);
102 yield
103 async_handshake(conn->next_layer(), *timer, std::move(self));
104 AEDIS_CHECK_OP0();
105 self.complete({});
106 }
107 }
108};
