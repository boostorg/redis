25 net::awaitable<void> client::reader()
26 {
27  // Writes and reads continuosly from the socket.
28  for (std::string buffer;;) {
29  // Notice this coro can get scheduled while the write operation
30  // in the writer is ongoing. so we have to check.
31  while (!std::empty(req_info_) && req_info_.front().size != 0) {
32  assert(!std::empty(requests_));
33  boost::system::error_code ec;
34  co_await
35  net::async_write(
36  socket_,
37  net::buffer(requests_.data(), req_info_.front().size),
38  net::redirect_error(net::use_awaitable, ec));
39 
40  requests_.erase(0, req_info_.front().size);
41  req_info_.front().size = 0;
42 
43  if (req_info_.front().cmds != 0)
44  break; // We must await the responses.
45 
46  req_info_.pop();
47  }
48 
49  do { // Keeps reading while there are no messages queued waiting to be sent.
50  do { // Consumes the responses to all commands in the request.
51  boost::system::error_code ec;
52  auto const t =
53  co_await async_read_type(socket_, net::dynamic_buffer(buffer),
54  net::redirect_error(net::use_awaitable, ec));
55  if (ec) {
56  stop_writer_ = true;
57  timer_.cancel();
58  co_return;
59  }
60 
61  if (t == type::push) {
62  auto adapter = [this](type t, std::size_t aggregate_size, std::size_t depth, char const* data, std::size_t size, std::error_code& ec)
63  {extended_adapter_(redis::command::unknown, t, aggregate_size, depth, data, size, ec);};
64 
65  co_await resp3::async_read(socket_, net::dynamic_buffer(buffer), adapter, net::redirect_error(net::use_awaitable, ec));
66  on_msg_(ec, redis::command::unknown);
67  if (ec) { // TODO: Return only on non aedis errors.
68  stop_writer_ = true;
69  timer_.cancel();
70  co_return;
71  }
72 
73  } else {
74  auto adapter = [this](type t, std::size_t aggregate_size, std::size_t depth, char const* data, std::size_t size, std::error_code& ec)
75  {extended_adapter_(commands_.front(), t, aggregate_size, depth, data, size, ec);};
76 
77  boost::system::error_code ec;
78  co_await resp3::async_read(socket_, net::dynamic_buffer(buffer), adapter, net::redirect_error(net::use_awaitable, ec));
79  on_msg_(ec, commands_.front());
80  if (ec) { // TODO: Return only on non aedis errors.
81  stop_writer_ = true;
82  timer_.cancel();
83  co_return;
84  }
85 
86  commands_.pop();
87  --req_info_.front().cmds;
88  }
89 
90  } while (!std::empty(req_info_) && req_info_.front().cmds != 0);
91 
92  // We may exit the loop above either because we are done
93  // with the response or because we received a server push
94  // while the queue was empty so we have to check before
95  // poping..
96  if (!std::empty(req_info_))
97  req_info_.pop();
98 
99  } while (std::empty(req_info_));
100  }
101 }

103 net::awaitable<void> client::writer()
104 {
105  boost::system::error_code ec;
106  while (socket_.is_open()) {
107  ec = {};
108  co_await timer_.async_wait(net::redirect_error(net::use_awaitable, ec));
109  if (stop_writer_)
110  co_return;
111 
112  // Notice this coro can get scheduled while the write operation
113  // in the reader is ongoing. so we have to check.
114  while (!std::empty(req_info_) && req_info_.front().size != 0) {
115  assert(!std::empty(requests_));
116  ec = {};
117  co_await net::async_write(
118  socket_, net::buffer(requests_.data(), req_info_.front().size),
119  net::redirect_error(net::use_awaitable, ec));
120  if (ec) {
121  // What should we do here exactly? Closing the socket will
122  // cause the reader coroutine to return so that the engage
123  // coroutine returns to the user.
124  socket_.close();
125  co_return;
126  }
127 
128  requests_.erase(0, req_info_.front().size);
129  req_info_.front().size = 0;
130 
131  if (req_info_.front().cmds != 0)
132  break;
133 
134  req_info_.pop();
135  }
136  }

139 bool client::prepare_next()
140 {
141  if (std::empty(req_info_)) {
142  req_info_.push({});
143  return true;
144  }
145 
146  if (req_info_.front().size == 0) {
147  // It has already been written and we are waiting for the
148  // responses.
149  req_info_.push({});
150  return false;
151  }
152 
153  return false;
154 }

166 net::awaitable<void> client::engage(socket_type socket)
167 {
168  using namespace aedis::net::experimental::awaitable_operators;
169 
170  socket_ = std::move(socket);
171 
172  std::string request;
173  auto sr = redis::make_serializer(request);
174  sr.push(redis::command::hello, 3);
175 
176  boost::system::error_code ec;
177  co_await net::async_write(socket_, net::buffer(request), net::redirect_error(net::use_awaitable, ec));
178  if (ec)
179  co_return;
180 
181  std::string buffer;
182  auto adapter = [this](type t, std::size_t aggregate_size, std::size_t depth, char const* data, std::size_t size, std::error_code& ec)
183  {extended_adapter_(redis::command::hello, t, aggregate_size, depth, data, size, ec);};
184 
185  co_await
186  resp3::async_read(
187  socket_, net::dynamic_buffer(buffer), adapter, net::redirect_error(net::use_awaitable, ec));
188 
189  on_msg_(ec, redis::command::hello);
190 
191  if (ec)
192  co_return;
193 
194  co_await (reader() && writer());
195 }
