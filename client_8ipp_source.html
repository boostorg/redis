23 net::awaitable<void> client::reader()
24 {
25  // Writes and reads continuosly from the socket.
26  for (std::string buffer;;) {
27  while (!std::empty(req_info_)) {
28  co_await net::async_write(socket_, net::buffer(requests_.data(), req_info_.front().size));
29 
30  requests_.erase(0, req_info_.front().size);
31 
32  if (req_info_.front().cmds != 0)
33  break; // We must await the responses.
34 
35  req_info_.pop();
36  }
37 
38  do { // Keeps reading while there are no messages queued waiting to be sent.
39  do { // Consumes the responses to all commands in the request.
40  auto const t = co_await async_read_type(socket_, net::dynamic_buffer(buffer));
41  if (t == type::push) {
42  auto adapter = [this](type t, std::size_t aggregate_size, std::size_t depth, char const* data, std::size_t size, std::error_code& ec)
43  {adapter_(redis::command::unknown, t, aggregate_size, depth, data, size, ec);};
44 
45  boost::system::error_code ec;
46  co_await resp3::async_read(socket_, net::dynamic_buffer(buffer), adapter, net::redirect_error(net::use_awaitable, ec));
47  on_msg_(ec, redis::command::unknown);
48  } else {
49  auto adapter = [this](type t, std::size_t aggregate_size, std::size_t depth, char const* data, std::size_t size, std::error_code& ec)
50  {adapter_(commands_.front(), t, aggregate_size, depth, data, size, ec);};
51 
52  boost::system::error_code ec;
53  co_await resp3::async_read(socket_, net::dynamic_buffer(buffer), adapter, net::redirect_error(net::use_awaitable, ec));
54  on_msg_(ec, commands_.front());
55  commands_.pop();
56  --req_info_.front().cmds;
57  }
58 
59  } while (!std::empty(req_info_) && req_info_.front().cmds != 0);
60 
61  // We may exit the loop above either because we are done
62  // with the response or because we received a server push
63  // while the queue was empty so we have to check before
64  // poping..
65  if (!std::empty(req_info_))
66  req_info_.pop();
67 
68  } while (std::empty(req_info_));
69  }
70 }

72 net::awaitable<void> client::writer()
73 {
74  while (socket_.is_open()) {
75  boost::system::error_code ec;
76  co_await timer_.async_wait(net::redirect_error(net::use_awaitable, ec));
77  while (!std::empty(req_info_)) {
78  co_await net::async_write(socket_, net::buffer(requests_.data(), req_info_.front().size));
79  requests_.erase(0, req_info_.front().size);
80 
81  if (req_info_.front().cmds != 0)
82  break;
83 
84  req_info_.pop();
85  }
86  }
87 }

102 net::awaitable<void>
103 client::connection_manager()
104 {
105  using namespace aedis::net::experimental::awaitable_operators;
106  using tcp_resolver = aedis::net::use_awaitable_t<>::as_default_on_t<aedis::net::ip::tcp::resolver>;
107 
108  for (;;) {
109  tcp_resolver resolver{socket_.get_executor()};
110  auto const res = co_await resolver.async_resolve("127.0.0.1", "6379");
111  co_await net::async_connect(socket_, res);
112 
113  co_await say_hello();
114 
115  timer_.expires_at(std::chrono::steady_clock::time_point::max());
116  co_await (reader() && writer());
117 
118  socket_.close();
119  timer_.cancel();
120 
121  timer_.expires_after(std::chrono::seconds{1});
122  boost::system::error_code ec;
123  co_await timer_.async_wait(net::redirect_error(net::use_awaitable, ec));
124  }
125 }
