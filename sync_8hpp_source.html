47  template <class ResponseAdapter>
48  std::size_t
49  exec(resp3::request const& req, ResponseAdapter adapter, boost::system::error_code& ec)
50  {
51  sync_helper sh;
52  std::size_t res = 0;
53 
54  auto f = [this, &ec, &res, &sh, &req, adapter]()
55  {
56  conn_.async_exec(req, adapter, [&sh, &res, &ec](auto const& ecp, std::size_t n) {
57  std::unique_lock ul(sh.mutex);
58  ec = ecp;
59  res = n;
60  sh.ready = true;
61  ul.unlock();
62  sh.cv.notify_one();
63  });
64  };
65 
66  boost::asio::dispatch(boost::asio::bind_executor(conn_.get_executor(), f));
67  std::unique_lock lk(sh.mutex);
68  sh.cv.wait(lk, [&sh]{return sh.ready;});
69  return res;
70  }
