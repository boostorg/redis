22template <class Channel>
23struct send_receive_op {
24 Channel* channel;
25 boost::asio::coroutine coro{};
26
27 template <class Self>
28 void operator()( Self& self
29 , boost::system::error_code ec = {}
30 , std::size_t = 0)
31 {
32 reenter (coro)
33 {
34 yield
35 channel->async_send(boost::system::error_code{}, 0, std::move(self));
36 AEDIS_CHECK_OP1();
37
38 yield
39 channel->async_receive(std::move(self));
40 AEDIS_CHECK_OP1();
41
42 self.complete({}, 0);
43 }
44 }
45};
46
47template <
48 class Channel,
49 class CompletionToken =
50 boost::asio::default_completion_token_t<typename Channel::executor_type>
51 >
52auto async_send_receive(Channel& channel, CompletionToken&& token = CompletionToken{})
53{
54 return boost::asio::async_compose
55 < CompletionToken
56 , void(boost::system::error_code, std::size_t)
57 >(send_receive_op<Channel>{&channel}, token, channel);
58}
