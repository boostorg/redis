22template <class Executor>
23using conn_timer_t = boost::asio::basic_waitable_timer<std::chrono::steady_clock, boost::asio::wait_traits<std::chrono::steady_clock>, Executor>;
24
25template <
26 class Stream,
27 class EndpointSequence
28 >
29struct connect_op {
30 Stream* socket;
31 conn_timer_t<typename Stream::executor_type>* timer;
32 EndpointSequence* endpoints;
33 boost::asio::coroutine coro{};
34
35 template <class Self>
36 void operator()( Self& self
37 , std::array<std::size_t, 2> order = {}
38 , boost::system::error_code ec1 = {}
39 , typename Stream::protocol_type::endpoint const& ep = {}
40 , boost::system::error_code ec2 = {})
41 {
42 reenter (coro)
43 {
44 yield
45 boost::asio::experimental::make_parallel_group(
46 [this](auto token)
47 {
48 auto f = [](boost::system::error_code const&, auto const&) { return true; };
49 return boost::asio::async_connect(*socket, *endpoints, f, token);
50 },
51 [this](auto token) { return timer->async_wait(token);}
52 ).async_wait(
53 boost::asio::experimental::wait_for_one(),
54 std::move(self));
55
56 if (is_cancelled(self)) {
57 self.complete(boost::asio::error::operation_aborted, {});
58 return;
59 }
60
61 switch (order[0]) {
62 case 0: self.complete(ec1, ep); return;
63 case 1:
64 {
65 if (ec2) {
66 self.complete(ec2, {});
67 } else {
68 self.complete(error::connect_timeout, ep);
69 }
70 return;
71 }
72
73 default: BOOST_ASSERT(false);
74 }
75 }
76 }
77};

79template <class Resolver, class Timer>
80struct resolve_op {
81 Resolver* resv;
82 Timer* timer;
83 boost::string_view host;
84 boost::string_view port;
85 boost::asio::coroutine coro{};
86
87 template <class Self>
88 void operator()( Self& self
89 , std::array<std::size_t, 2> order = {}
90 , boost::system::error_code ec1 = {}
91 , boost::asio::ip::tcp::resolver::results_type res = {}
92 , boost::system::error_code ec2 = {})
93 {
94 reenter (coro)
95 {
96 yield
97 boost::asio::experimental::make_parallel_group(
98 [this](auto token) { return resv->async_resolve(host.data(), port.data(), token);},
99 [this](auto token) { return timer->async_wait(token);}
100 ).async_wait(
101 boost::asio::experimental::wait_for_one(),
102 std::move(self));
103
104 if (is_cancelled(self)) {
105 self.complete(boost::asio::error::operation_aborted, {});
106 return;
107 }
108
109 switch (order[0]) {
110 case 0: self.complete(ec1, res); return;
111
112 case 1:
113 {
114 if (ec2) {
115 self.complete(ec2, {});
116 } else {
117 self.complete(error::resolve_timeout, {});
118 }
119 return;
120 }
121
122 default: BOOST_ASSERT(false);
123 }
124 }
125 }
126};

128template <class Channel>
129struct send_receive_op {
130 Channel* channel;
131 boost::asio::coroutine coro{};
132
133 template <class Self>
134 void operator()( Self& self
135 , boost::system::error_code ec = {}
136 , std::size_t = 0)
137 {
138 reenter (coro)
139 {
140 yield
141 channel->async_send(boost::system::error_code{}, 0, std::move(self));
142 AEDIS_CHECK_OP1();
143
144 yield
145 channel->async_receive(std::move(self));
146 AEDIS_CHECK_OP1();
147
148 self.complete({}, 0);
149 }
150 }
151};

153template <
154 class Stream,
155 class EndpointSequence,
156 class CompletionToken
157 >
158auto async_connect(
159 Stream& socket,
160 conn_timer_t<typename Stream::executor_type>& timer,
161 EndpointSequence ep,
162 CompletionToken&& token)
163{
164 return boost::asio::async_compose
165 < CompletionToken
166 , void(boost::system::error_code, typename Stream::protocol_type::endpoint const&)
167 >(connect_op<Stream, EndpointSequence>
168 {&socket, &timer, &ep}, token, socket, timer);
169}

171template <
172 class Resolver,
173 class Timer,
174 class CompletionToken =
175 boost::asio::default_completion_token_t<typename Resolver::executor_type>
176 >
177auto async_resolve(
178 Resolver& resv,
179 Timer& timer,
180 boost::string_view host,
181 boost::string_view port,
182 CompletionToken&& token = CompletionToken{})
183{
184 return boost::asio::async_compose
185 < CompletionToken
186 , void(boost::system::error_code, boost::asio::ip::tcp::resolver::results_type)
187 >(resolve_op<Resolver, Timer>{&resv, &timer, host, port}, token, resv, timer);
188}

190template <
191 class Channel,
192 class CompletionToken =
193 boost::asio::default_completion_token_t<typename Channel::executor_type>
194 >
195auto async_send_receive(Channel& channel, CompletionToken&& token = CompletionToken{})
196{
197 return boost::asio::async_compose
198 < CompletionToken
199 , void(boost::system::error_code, std::size_t)
200 >(send_receive_op<Channel>{&channel}, token, channel);
201}
