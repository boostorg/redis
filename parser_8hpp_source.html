21 // Converts a wire-format RESP3 type (char) to a resp3 type.
22 type to_type(char c);

24 template <class ResponseAdapter>
25 class parser {
26 private:
27  static constexpr std::size_t max_embedded_depth = 5;
28 
29  ResponseAdapter adapter_;
30 
31  // The current depth. Simple data types will have depth 0, whereas
32  // the elements of aggregates will have depth 1. Embedded types
33  // will have increasing depth.
34  std::size_t depth_ = 0;
35 
36  // The parser supports up to 5 levels of nested structures. The
37  // first element in the sizes stack is a sentinel and must be
38  // different from 1.
39  std::size_t sizes_[max_embedded_depth + 1] = {1};
40 
41  // Contains the length expected in the next bulk read.
42  std::size_t bulk_length_ = (std::numeric_limits<std::size_t>::max)();
43 
44  // The type of the next bulk. Contains type::invalid if no bulk is
45  // expected.
46  type bulk_ = type::invalid;
47 
48 public:
49  parser(ResponseAdapter adapter)
50  : adapter_{adapter}
51  {
52  sizes_[0] = 2; // The sentinel must be more than 1.
53  }
54 
55  // Returns the number of bytes that have been consumed.
56  std::size_t
57  advance(char const* data, std::size_t n, std::error_code& ec)
58  {
59  if (bulk_ != type::invalid) {
60  n = bulk_length_ + 2;
61  switch (bulk_) {
62  case type::streamed_string_part:
63  {
64  if (bulk_length_ == 0) {
65  sizes_[depth_] = 1;
66  } else {
67  adapter_(bulk_, 1, depth_, data, bulk_length_, ec);
68  if (ec)
69  return 0;
70  }
71  } break;
72  default:
73  {
74  adapter_(bulk_, 1, depth_, data, bulk_length_, ec);
75  if (ec)
76  return 0;
77  }
78  }
79 
80  bulk_ = type::invalid;
81  --sizes_[depth_];
82 
83  } else if (sizes_[depth_] != 0) {
84  auto const t = to_type(*data);
85  switch (t) {
86  case type::blob_error:
87  case type::verbatim_string:
88  case type::streamed_string_part:
89  {
90  auto const r =
91  std::from_chars(data + 1, data + n - 2, bulk_length_);
92  if (r.ec != std::errc()) {
93  ec = error::not_a_number;
94  return 0;
95  }
96 
97  bulk_ = t;
98  } break;
99  case type::blob_string:
100  {
101  if (*(data + 1) == '?') {
102  // Trick: A streamed string is read as an aggregate
103  // of infinite lenght. When the streaming is done
104  // the server is supposed to send a part with lenght
105  // 0.
106  sizes_[++depth_] = (std::numeric_limits<std::size_t>::max)();
107  } else {
108  auto const r =
109  std::from_chars(data + 1, data + n - 2, bulk_length_);
110  if (r.ec != std::errc()) {
111  ec = error::not_a_number;
112  return 0;
113  }
114 
115  bulk_ = type::blob_string;
116  }
117  } break;
118  case type::simple_error:
119  case type::number:
120  case type::doublean:
121  case type::boolean:
122  case type::big_number:
123  case type::simple_string:
124  {
125  adapter_(t, 1, depth_, data + 1, n - 3, ec);
126  if (ec)
127  return 0;
128 
129  --sizes_[depth_];
130  } break;
131  case type::null:
132  {
133  adapter_(type::null, 1, depth_, nullptr, 0, ec);
134  if (ec)
135  return 0;
136 
137  --sizes_[depth_];
138  } break;
139  case type::push:
140  case type::set:
141  case type::array:
142  case type::attribute:
143  case type::map:
144  {
145  std::size_t l;
146  auto const r = std::from_chars(data + 1, data + n - 2, l);
147  if (r.ec != std::errc()) {
148  ec = error::not_a_number;
149  return 0;
150  }
151 
152  adapter_(t, l, depth_, nullptr, 0, ec);
153  if (ec)
154  return 0;
155 
156  if (l == 0) {
157  --sizes_[depth_];
158  } else {
159  if (depth_ == max_embedded_depth) {
160  ec = error::exceeeds_max_nested_depth;
161  return 0;
162  }
163 
164  ++depth_;
165 
166  sizes_[depth_] = l * element_multiplicity(t);
167  }
168  } break;
169  default:
170  {
171  ec = error::invalid_type;
172  return 0;
173  }
174  }
175  }
176 
177  while (sizes_[depth_] == 0) {
178  --depth_;
179  --sizes_[depth_];
180  }
181 
182  return n;
183  }
184 
185  // Returns true when the parser is done with the current message.
186  auto done() const noexcept
187  { return depth_ == 0 && bulk_ == type::invalid; }
188 
189  // The bulk type expected in the next read. If none is expected returns
190  // type::invalid.
191  auto bulk() const noexcept { return bulk_; }
192 
193  // The length expected in the the next bulk.
194  auto bulk_length() const noexcept { return bulk_length_; }
195 };
