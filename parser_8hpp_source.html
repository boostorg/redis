22auto parse_uint(char const* data, std::size_t size, boost::system::error_code& ec) -> std::size_t;
23
24template <class ResponseAdapter>
25class parser {
26private:
27 using node_type = node<boost::string_view>;
28 static constexpr std::size_t max_embedded_depth = 5;
29
30 ResponseAdapter adapter_;
31
32 // The current depth. Simple data types will have depth 0, whereas
33 // the elements of aggregates will have depth 1. Embedded types
34 // will have increasing depth.
35 std::size_t depth_ = 0;
36
37 // The parser supports up to 5 levels of nested structures. The
38 // first element in the sizes stack is a sentinel and must be
39 // different from 1.
40 std::array<std::size_t, max_embedded_depth + 1> sizes_ = {{1}};
41
42 // Contains the length expected in the next bulk read.
43 std::size_t bulk_length_ = (std::numeric_limits<std::size_t>::max)();
44
45 // The type of the next bulk. Contains type::invalid if no bulk is
46 // expected.
47 type bulk_ = type::invalid;
48
49public:
50 explicit parser(ResponseAdapter adapter)
51 : adapter_{adapter}
52 {
53 sizes_[0] = 2; // The sentinel must be more than 1.
54 }
55
56 // Returns the number of bytes that have been consumed.
57 auto
58 consume(char const* data, std::size_t n, boost::system::error_code& ec) -> std::size_t
59 {
60 if (bulk_ != type::invalid) {
61 n = bulk_length_ + 2;
62 switch (bulk_) {
63 case type::streamed_string_part:
64 {
65 BOOST_ASSERT(bulk_length_ != 0);
66 adapter_({bulk_, 1, depth_, {data, bulk_length_}}, ec);
67 if (ec)
68 return 0;
69 } break;
70 default:
71 {
72 adapter_({bulk_, 1, depth_, {data, bulk_length_}}, ec);
73 if (ec)
74 return 0;
75 }
76 }
77
78 bulk_ = type::invalid;
79 --sizes_[depth_];
80
81 } else if (sizes_[depth_] != 0) {
82 auto const t = to_type(*data);
83 switch (t) {
84 case type::streamed_string_part:
85 {
86 bulk_length_ = parse_uint(data + 1, n - 2, ec);
87 if (ec)
88 return 0;
89
90 if (bulk_length_ == 0) {
91 adapter_({type::streamed_string_part, 1, depth_, {}}, ec);
92 sizes_[depth_] = 0; // We are done.
93 } else {
94 bulk_ = type::streamed_string_part;
95 }
96 } break;
97 case type::blob_error:
98 case type::verbatim_string:
99 case type::blob_string:
100 {
101 if (data[1] == '?') {
102 // NOTE: This can only be triggered with blob_string.
103 // Trick: A streamed string is read as an aggregate
104 // of infinite lenght. When the streaming is done
105 // the server is supposed to send a part with length
106 // 0.
107 sizes_[++depth_] = (std::numeric_limits<std::size_t>::max)();
108 } else {
109 bulk_length_ = parse_uint(data + 1, n - 2, ec);
110 if (ec)
111 return 0;
112
113 bulk_ = t;
114 }
115 } break;
116 case type::boolean:
117 {
118 if (n == 3) {
119 ec = error::empty_field;
120 return 0;
121 }
122
123 if (data[1] != 'f' && data[1] != 't') {
124 ec = error::unexpected_bool_value;
125 return 0;
126 }
127
128 adapter_({t, 1, depth_, {data + 1, n - 3}}, ec);
129 if (ec)
130 return 0;
131
132 --sizes_[depth_];
133 } break;
134 case type::doublean:
135 case type::big_number:
136 case type::number:
137 {
138 if (n == 3) {
139 ec = error::empty_field;
140 return 0;
141 }
142
143 adapter_({t, 1, depth_, {data + 1, n - 3}}, ec);
144 if (ec)
145 return 0;
146
147 --sizes_[depth_];
148 } break;
149 case type::simple_error:
150 case type::simple_string:
151 {
152 adapter_({t, 1, depth_, {&data[1], n - 3}}, ec);
153 if (ec)
154 return 0;
155
156 --sizes_[depth_];
157 } break;
158 case type::null:
159 {
160 adapter_({type::null, 1, depth_, {}}, ec);
161 if (ec)
162 return 0;
163
164 --sizes_[depth_];
165 } break;
166 case type::push:
167 case type::set:
168 case type::array:
169 case type::attribute:
170 case type::map:
171 {
172 auto const l = parse_uint(data + 1, n - 2, ec);
173 if (ec)
174 return 0;
175
176 adapter_({t, l, depth_, {}}, ec);
177 if (ec)
178 return 0;
179
180 if (l == 0) {
181 --sizes_[depth_];
182 } else {
183 if (depth_ == max_embedded_depth) {
184 ec = error::exceeeds_max_nested_depth;
185 return 0;
186 }
187
188 ++depth_;
189
190 sizes_[depth_] = l * element_multiplicity(t);
191 }
192 } break;
193 default:
194 {
195 ec = error::invalid_data_type;
196 return 0;
197 }
198 }
199 }
200
201 while (sizes_[depth_] == 0) {
202 --depth_;
203 --sizes_[depth_];
204 }
205
206 return n;
207 }
208
209 // Returns true when the parser is done with the current message.
210 [[nodiscard]] auto done() const noexcept
211 { return depth_ == 0 && bulk_ == type::invalid; }
212
213 // The bulk type expected in the next read. If none is expected returns
214 // type::invalid.
215 auto bulk() const noexcept { return bulk_; }
216
217 // The length expected in the the next bulk.
218 [[nodiscard]] auto bulk_length() const noexcept { return bulk_length_; }
219};
