19using int_type = std::uint64_t;
20
21class parser {
22private:
23 using node_type = node<std::string_view>;
24 static constexpr std::size_t max_embedded_depth = 5;
25
26 // The current depth. Simple data types will have depth 0, whereas
27 // the elements of aggregates will have depth 1. Embedded types
28 // will have increasing depth.
29 std::size_t depth_ = 0;
30
31 // The parser supports up to 5 levels of nested structures. The
32 // first element in the sizes stack is a sentinel and must be
33 // different from 1.
34 std::array<std::size_t, max_embedded_depth + 1> sizes_ = {{1}};
35
36 // Contains the length expected in the next bulk read.
37 int_type bulk_length_ = (std::numeric_limits<unsigned long>::max)();
38
39 // The type of the next bulk. Contains type::invalid if no bulk is
40 // expected.
41 type bulk_ = type::invalid;
42
43public:
44 parser();
45
46 // Returns the number of bytes that have been consumed.
47 auto
48 consume(
49 char const* data,
50 std::size_t n,
51 boost::system::error_code& ec) -> std::pair<node_type, std::size_t>;
52
53 // Returns true when the parser is done with the current message.
54 [[nodiscard]] auto done() const noexcept
55 { return depth_ == 0 && bulk_ == type::invalid; }
56
57 // The bulk type expected in the next read. If none is expected
58 // returns type::invalid.
59 [[nodiscard]] auto bulk_expected() const noexcept -> bool
60 { return bulk_ != type::invalid; }
61
62 // The length expected in the the next bulk.
63 [[nodiscard]] auto bulk_length() const noexcept
64 { return bulk_length_; }
65};
