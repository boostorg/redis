35 template <
36  class SyncReadStream,
37  class DynamicBuffer,
38  class ResponseAdapter
39  >
40 std::size_t
41 read(
42  SyncReadStream& stream,
43  DynamicBuffer buf,
44  ResponseAdapter adapter,
45  boost::system::error_code& ec)
46 {
47  detail::parser p {adapter};
48  std::size_t n = 0;
49  std::size_t consumed = 0;
50  do {
51  if (p.bulk() == type::invalid) {
52  n = net::read_until(stream, buf, "\r\n", ec);
53  if (ec)
54  return 0;
55 
56  if (n < 3) {
57  ec = error::unexpected_read_size;
58  return 0;
59  }
60  } else {
61  auto const s = std::size(buf);
62  auto const l = p.bulk_length();
63  if (s < (l + 2)) {
64  auto const to_read = l + 2 - s;
65  buf.grow(to_read);
66  n = net::read(stream, buf.data(s, to_read), ec);
67  if (ec)
68  return 0;
69 
70  if (n < to_read) {
71  ec = error::unexpected_read_size;
72  return 0;
73  }
74  }
75  }
76 
77  std::error_code ec;
78  auto const* data = (char const*) buf.data(0, n).data();
79  n = p.advance(data, n, ec);
80  if (ec)
81  return 0;
82 
83  buf.consume(n);
84  consumed += n;
85  } while (!p.done());
86 
87  return consumed;
88 }
