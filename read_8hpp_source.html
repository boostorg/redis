50template <
51 class SyncReadStream,
52 class DynamicBuffer,
53 class ResponseAdapter
54 >
55auto
56read(
57 SyncReadStream& stream,
58 DynamicBuffer buf,
59 ResponseAdapter adapter,
60 boost::system::error_code& ec) -> std::size_t
61{
62 detail::parser p;
63 std::size_t n = 0;
64 std::size_t consumed = 0;
65 do {
66 if (!p.bulk_expected()) {
67 n = boost::asio::read_until(stream, buf, "\r\n", ec);
68 if (ec)
69 return 0;
70
71 } else {
72 auto const s = buf.size();
73 auto const l = p.bulk_length();
74 if (s < (l + 2)) {
75 auto const to_read = l + 2 - s;
76 buf.grow(to_read);
77 n = boost::asio::read(stream, buf.data(s, to_read), ec);
78 if (ec)
79 return 0;
80 }
81 }
82
83 auto const* data = static_cast<char const*>(buf.data(0, n).data());
84 auto const res = p.consume(data, n, ec);
85 if (ec)
86 return 0;
87
88 if (!p.bulk_expected()) {
89 adapter(res.first, ec);
90 if (ec)
91 return 0;
92 }
93
94 buf.consume(res.second);
95 consumed += res.second;
96 } while (!p.done());
97
98 return consumed;
99}
