54 template <
55  class SyncReadStream,
56  class DynamicBuffer,
57  class ResponseAdapter
58  >
59 std::size_t
60 read(
61  SyncReadStream& stream,
62  DynamicBuffer buf,
63  ResponseAdapter adapter,
64  boost::system::error_code& ec)
65 {
66  detail::parser<ResponseAdapter> p {adapter};
67  std::size_t n = 0;
68  std::size_t consumed = 0;
69  do {
70  if (p.bulk() == type::invalid) {
71  n = boost::asio::read_until(stream, buf, "\r\n", ec);
72  if (ec)
73  return 0;
74 
75  if (n < 3) {
76  ec = error::unexpected_read_size;
77  return 0;
78  }
79  } else {
80  auto const s = buf.size();
81  auto const l = p.bulk_length();
82  if (s < (l + 2)) {
83  auto const to_read = l + 2 - s;
84  buf.grow(to_read);
85  n = boost::asio::read(stream, buf.data(s, to_read), ec);
86  if (ec)
87  return 0;
88 
89  if (n < to_read) {
90  ec = error::unexpected_read_size;
91  return 0;
92  }
93  }
94  }
95 
96  auto const* data = (char const*) buf.data(0, n).data();
97  n = p.consume(data, n, ec);
98  if (ec)
99  return 0;
100 
101  buf.consume(n);
102  consumed += n;
103  } while (!p.done());
104 
105  return consumed;
106 }
