86int main()

91 std::set<user> users

93
94 request req;
95 req.push("HELLO", 3);
96 req.push_range("SADD", "sadd-key", users); // Sends
97 req.push("SMEMBERS", "sadd-key"); // Retrieves
98 req.push("QUIT");
99
100 std::tuple<aedis::ignore, int, std::set<user>, std::string> resp;
101
102 endpoint ep{"127.0.0.1", "6379"};
103 db.async_run(ep, req, adapt(resp), {}, [](auto ec, auto) {
104 std::cout << ec.message() << std::endl;
105 });
106
107 ioc.run();
108
109 // Print
110 print(std::get<2>(resp));
111}
