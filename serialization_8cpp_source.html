72  boost::system::error_code& ec)
73  {
74  switch (cmd) {
75  case command::get: adapter::get<boost::optional<mystruct>>(adapters_)(nd, ec); break;
76  case command::lrange: adapter::get<std::list<mystruct>>(adapters_)(nd, ec); break;
77  case command::smembers: adapter::get<std::set<mystruct>>(adapters_)(nd, ec); break;
78  case command::hgetall: adapter::get<std::map<std::string, mystruct>>(adapters_)(nd, ec); break;

83  void on_read(command cmd)
84  {
85  std::cout << cmd << "\n";
86 
87  switch (cmd) {
88  case command::hello:
89  {
90  mystruct var{1, 2};
91 
92  std::map<std::string, mystruct> map
93  { {"key1", {1, 2}}
94  , {"key2", {3, 4}}
95  , {"key3", {5, 6}}};
96 
97  std::vector<mystruct> vec
98  {{1, 2}, {3, 4}, {5, 6}};
99 
100  std::set<std::string> set
101  {{1, 2}, {3, 4}, {5, 6}};
102 
103  // Sends
104  db_->send(command::set, "serialization-var-key", var, "EX", "2");
105  db_->send_range(command::hset, "serialization-hset-key", map);
106  db_->send_range(command::rpush, "serialization-rpush-key", vec);
107  db_->send_range(command::sadd, "serialization-sadd-key", set);
108 
109  // Retrieves
110  db_->send(command::get, "serialization-var-key");
111  db_->send(command::hgetall, "serialization-hset-key");
112  db_->send(command::lrange, "serialization-rpush-key", 0, -1);
113  db_->send(command::smembers, "serialization-sadd-key");
114  } break;
115 
116  case command::get:
117  {
118  if (std::get<boost::optional<mystruct>>(resps_).has_value()) {
119  std::cout << std::get<boost::optional<mystruct>>(resps_).value() << "\n\n";
120  std::get<boost::optional<mystruct>>(resps_).reset();
121  } else {
122  std::cout << "Expired." << "\n";
123  }
124  } break;
125 
126  case command::lrange:
127  for (auto const& e: std::get<std::list<mystruct>>(resps_))
128  std::cout << e << "\n";
129  std::cout << "\n";
130  std::get<std::list<mystruct>>(resps_).clear();
131  break;
132 
133  case command::smembers:
134  for (auto const& e: std::get<std::set<mystruct>>(resps_))
135  std::cout << e << "\n";
136  std::cout << "\n";
137  std::get<std::set<mystruct>>(resps_).clear();
138  break;
139 
140  case command::hgetall:
141  for (auto const& e: std::get<std::map<std::string, mystruct>>(resps_))
142  std::cout << e.first << ", " << e.second << std::endl;
143  std::cout << "\n";
144  std::get<std::map<std::string, mystruct>>(resps_).clear();
145  break;
146 
147  default:;
148  }
149  }
