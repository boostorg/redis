25 /* This example will subscribe and read pushes indefinitely.
26  *
27  * To test send messages with redis-cli
28  *
29  * $ redis-cli -3
30  * 127.0.0.1:6379> PUBLISH channel some-message
31  * (integer) 3
32  * 127.0.0.1:6379>
33  *
34  * To test reconnection try, for example, to close all clients currently
35  * connected to the Redis instance
36  *
37  * $ redis-cli
38  * > CLIENT kill TYPE pubsub
39  */

41 // Receives pushes.
42 net::awaitable<void> push_receiver(std::shared_ptr<connection> db)
43 {
44  for (std::vector<node<std::string>> resp;;) {
45  co_await db->async_receive_push(adapt(resp));
46  print_push(resp);
47  resp.clear();
48  }
49 }
50 
51 // Receives events.
52 net::awaitable<void> event_receiver(std::shared_ptr<connection> db)
53 {
54  request req;
55  req.push("SUBSCRIBE", "channel");
56 
57  for (;;) {
58  auto ev = co_await db->async_receive_event();
59  if (ev == connection::event::hello)
60  co_await db->async_exec(req);
61  }
62 }

64 int main()
65 {
66  try {
67  net::io_context ioc;
68  auto db = std::make_shared<connection>(ioc);
69 
70  db->get_config().enable_events = true;
71  db->get_config().enable_reconnect = true;
72 
73  net::co_spawn(ioc, push_receiver(db), net::detached);
74  net::co_spawn(ioc, event_receiver(db), net::detached);
75  db->async_run(net::detached);
76 
77  net::signal_set signals(ioc, SIGINT, SIGTERM);
78  signals.async_wait([&](auto, auto){ ioc.stop(); });
79 
80  ioc.run();
81  } catch (std::exception const& e) {
82  std::cerr << "Error: " << e.what() << std::endl;
83  }
