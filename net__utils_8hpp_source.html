10 using tcp_socket = aedis::net::use_awaitable_t<>::as_default_on_t<aedis::net::ip::tcp::socket>;
11 using tcp_resolver = aedis::net::use_awaitable_t<>::as_default_on_t<aedis::net::ip::tcp::resolver>;
12 using timer = aedis::net::use_awaitable_t<>::as_default_on_t<aedis::net::steady_timer>;
13 
14 aedis::net::awaitable<tcp_socket>
15 connect(
16  std::string host = "127.0.0.1",
17  std::string port = "6379")
18 {
19  auto ex = co_await aedis::net::this_coro::executor;
20  tcp_resolver resolver{ex};
21  auto const res = co_await resolver.async_resolve(host, port);
22  tcp_socket socket{ex};
23  co_await aedis::net::async_connect(socket, res);
24  co_return std::move(socket);
25 }
